# More LR - convert base location ranks into the expanded set on game start
on_game_start = {
	effect = {
		every_province = {
			every_location_in_province = {
				if = {
					limit = { location_rank = location_rank:city }
					change_location_rank = location_rank:major_city
				}
				else_if = {
					limit = { location_rank = location_rank:town }
					change_location_rank = location_rank:large_town
				}
				else_if = {
					limit = { location_rank = location_rank:rural_settlement }
					change_location_rank = location_rank:wilderness
				}
			}
		}
	}
}
