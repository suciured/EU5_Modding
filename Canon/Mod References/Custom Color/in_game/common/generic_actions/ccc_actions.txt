ccc_copy_country_color = {
	type = situation
	potential = {
		always = yes
		scope:actor = { var:ccc_color_page = 0 }
	}

	show_message = no
	show_message_to_target = no

	select_trigger = {
		looking_for_a = country
		target_flag = target_country
		name = "ccc_copy_country_color_select_country"
		none_available_msg_key = "ccc_copy_country_color_no_countries"
		column = { data = name }
		visible = { always = yes }
		enabled = { always = yes }
		
	}

	allow = { always = yes }

	effect = {
		hidden_effect = {
			scope:actor = {
				if = {
					limit = { NOT = { has_variable = ccc_saved_original } }
					if = {
						limit = { exists = c:CCB }
						c:CCB = { copy_country_color = scope:actor }
						set_variable = { name = ccc_saved_original value = yes }
					}
				}

				if = {
					limit = { exists = scope:target_country }
					copy_country_color = scope:target_country
					refresh_map_colors = yes
				}
			}
		}
	}

	ai_tick = never
	automation_tick = never
	ai_will_do = { base = 0 }
}
