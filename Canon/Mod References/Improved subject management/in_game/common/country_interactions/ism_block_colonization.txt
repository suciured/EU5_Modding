block_subject_colonization = {
	type = subject
	category = CATEGORY_SUBJECT_ACTIONS

	potential = {
	}
	
	allow = {
	}
	
	ai_tick = never

	use_enroute = no
	
	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				every_subject = {
					add_to_list = source
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			NOR = {
				country_type = building
				country_type = pop
			}
			is_subject_type = colonial_nation
			NOT = { has_country_modifier = overlord_blocking_colonization }
		}
        enabled = {
		}
	}

	effect = {
		if = {
			limit = {
				exists = scope:recipient
			}
			scope:recipient = {
				add_country_modifier = {
					modifier = overlord_blocking_colonization
					days = -1
					mode = replace
				}
			}
		}
	}
}
