move_subject_capital = { 
	type = subject
	category = CATEGORY_SUBJECT_ACTIONS

	ai_tick = never
	
	use_enroute = no

	potential = {
	}

	allow = {
	}

	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				every_subject = {
					add_to_list = source
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
            NOR = {
                country_type = building
                country_type = pop
            }
		}
        enabled = {
		}
	}

	select_trigger = {
		looking_for_a = location
		source = recipient
		target_flag = target
		source_flags = only_actual_locations
		name = "select_a_location_to_move_capital"
		none_available_msg_key = "no_valid_locations"
		column = {
			data = name
		}
		column = {
			data = population
		}
		column = {
			data = raw_material
		}
		column = {
			data = development
		}
		visible = {
			is_capital = no
		}
	}
	
	
	effect = {
		if = {
			limit = {
				exists = scope:target
			}
			scope:recipient = {
				set_capital = scope:target
			}
		}
		else = {
			custom_tooltip = select_a_location_to_move_capital
		}
	}
	
}
