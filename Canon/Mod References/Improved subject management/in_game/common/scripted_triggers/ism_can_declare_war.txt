# Prevent colonial subjects from declaring war on each other.
# And to only declaring war on nations located within the same continent.
REPLACE:colonial_nation_can_declare_wars_trigger = {
    scope:attacker ?= {
        AND = {
            exists = scope:attacker
            exists = scope:defender
            NOT = { scope:attacker.top_overlord ?= scope:defender.top_overlord }
            scope:defender.capital.continent = scope:attacker.capital.continent
        }
    }
}
