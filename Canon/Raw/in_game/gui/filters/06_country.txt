# COUNTRY
# root is player
# scope:target is the country to filter

country_only_within_diplomatic_range = {
	scope = country
	tag = diplomacy
	enabled_at_start = yes
	group = 1
	trigger = {
		exists = scope:target
		scope:target = {
			within_diplomatic_range = root
		}
	}
}

country_only_relevant_countries = {
	scope = country
	tag = diplomacy
	group = 2
	trigger = {
		exists = scope:target
		relevant_countries = scope:target
	}
}

country_only_neighbors = {
	scope = country
	tag = diplomacy
	group = 3
	trigger = {
		exists = scope:target
		is_neighbor_of = scope:target
	}
}

country_only_region = {
	scope = country
	tag = diplomacy
	group = 3
	trigger = {
		capital = {
			region = scope:target.capital.region
		}
	}
}

country_only_rivals = {
	scope = country
	tag = diplomacy
	group = 4
	trigger = {
		exists = scope:target
		is_rival_of = scope:target
	}
}

country_only_enemies = {
	scope = country
	tag = diplomacy
	group = 4
	trigger = {
		exists = scope:target
		is_enemy_of = scope:target
	}
}

country_only_positive_opinion = {
	scope = country
	tag = diplomacy
	group = 4
	trigger = {
		exists = scope:target
		has_positive_opinion = scope:target
	}
}

country_only_negative_opinion = {
	scope = country
	tag = diplomacy
	group = 4
	trigger = {
		exists = scope:target
		not = {has_positive_opinion = scope:target}
	}
}
country_is_great_power = {
	scope = country
	tag = diplomacy
	group = 5
	trigger = {
		is_great_power = yes
	}
}
