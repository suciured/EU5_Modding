namespace = hindu_events
hindu_events.1 = {
	hide_portraits = yes
	type = country_event
	title = hindu_events.1.title
	desc = hindu_events.1.desc

	trigger = {
		religion = religion:hindu
		any_international_organizations_member_of = {
			international_organization_type = international_organization_type:hindu_branch
			international_organization_has_policy = policy:smartism
		}
		has_heir = no
		has_ruler_in_country = yes
		government_type = government_type:monarchy
		NOT = { has_variable = has_had_suryavansha_flag }
		ruler = { has_dynasty = yes }
		any_character = {
			has_estate = estate_type:nobles_estate
			has_dynasty = yes
			dynasty = root.ruler.dynasty
			is_married = no
			age_in_years < 40
			is_female = yes
		}
	}

	illustration_tags = {
		10 = interior
		10 = happy
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:nobles_estate }
		ruler = { save_scope_as = target_ruler }
		set_variable = { name = has_had_suryavansha_flag value = 1 years = 100 }
		random_character = {
			limit = {
				has_estate = estate_type:nobles_estate
				has_dynasty = yes
				dynasty = root.ruler.dynasty
				is_married = no
				age_in_years < 40
				is_female = yes
			}
			save_scope_as = target_character
		}
		create_character = {
			age = 0
			mother = scope:target_character
			culture = scope:target_character.culture
			religion = scope:target_character.religion
			adm = { 5 10 }
			dip = { 5 10 }
			mil = { 5 10 }
			save_scope_as = target_child
		}
	}

	option = {
		name = hindu_events.1.a
		add_government_power = government_power_mild_penalty
		scope:target_ruler = {
			adopt_character = scope:target_child
		}
		set_as_designated_heir = scope:target_child
	}
	option = {
		name = hindu_events.1.b
		add_prestige = prestige_mild_penalty
	}
}

hindu_events.2 = {
	type = country_event
	title = hindu_events.2.title
	desc = hindu_events.2.desc

	trigger = {
		religion = religion:hindu
		at_war = yes
		has_ruler = yes
		has_heir = yes
		heir = { is_general = yes }
		any_international_organizations_member_of = {
			international_organization_type = international_organization_type:hindu_branch
			international_organization_has_policy = policy:shaktism
		}
		ruler = {
			any_child = {
				is_heir = no
				is_female = yes
				is_adult = yes
				age_in_years < 30
				NOT = { has_character_modifier = bala_tripurasundari }
			}
		}
		any_army = {
			has_commander = no
		}
	}

	illustration_tags = {
		10 = exterior
		10 = armed
	}

	immediate = {
		event_illustration_government_estate_effect = yes
		ruler = { 
			save_scope_as = target_ruler
			random_child = {
				limit = {
					is_heir = no
					is_female = yes
					is_adult = yes
					age_in_years < 30
					NOT = { has_character_modifier = bala_tripurasundari }
				}
				save_scope_as = target_daughter
			}
		}
		heir = { save_scope_as = target_heir }
	}

	option = {
		name = hindu_events.2.a
		add_prestige = prestige_mild_bonus
		scope:target_daughter = { 
			add_character_modifier = { modifier = bala_tripurasundari months = -1 mode = add_and_extend }
		}
		random_army = {
			limit = {
				has_commander = no
			}
			set_as_commander = scope:target_daughter
		}
	}
	option = {
		name = hindu_events.2.b
		add_prestige = prestige_mild_penalty
	}
}

scripted_trigger ruler_can_withdraw_with_god = {
	religion = religion:hindu
	has_ruler = yes
	ruler ?= { 
		NOT = { has_character_modifier = jnana }
		NOT = { has_character_modifier = hindu_withdrawn_contemplation }
	}
}

scripted_trigger ruler_can_withdraw_with_shiva = {
	ruler_can_withdraw_with_god = yes
	any_international_organizations_member_of = {
		international_organization_type = international_organization_type:hindu_branch
		OR = {
			international_organization_has_policy = policy:shaivism
			AND = {
				international_organization_has_policy = policy:smartism
				root = { is_favoring_god = { god = shiva_god } }
			}
		}
	}
}
