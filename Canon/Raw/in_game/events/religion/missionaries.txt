namespace = missionaries
missionaries.1 = {
	type = country_event
	title = missionaries.1.title
	desc = missionaries.1.desc
	trigger = {
	
		any_non_state_religion_location = {
			monthly_conversion > 0
			NOT = {	has_location_modifier = missionary_falls_ill	}
		}
	}
	
	immediate = {
		random_non_state_religion_location = {
			limit = {
				monthly_conversion > 0
				NOT = {	has_location_modifier = missionary_falls_ill	}
			}
			save_scope_as = target_location
		}
	}

	option = {
		name = missionaries.1.a
		add_government_power = government_power_extreme_penalty
		change_societal_value = { type = mysticism_vs_jurisprudence value = societal_value_tiny_move_to_right }
	}
	option = {
		name = missionaries.1.b
		scope:target_location = { add_location_modifier = { modifier = missionary_falls_ill months = 60 mode = add_and_extend } }
		change_societal_value = { type = mysticism_vs_jurisprudence value = societal_value_tiny_move_to_left }
	}
}

missionaries.2 = {
	type = country_event
	title = missionaries.2.title
	desc = missionaries.2.desc
	trigger = {
	
		any_non_state_religion_location = {
			monthly_conversion > 0
			is_core_of = root
			NOT = {	has_location_modifier = missionary_failure	}
			any_neighbor_location = {
				owner ?= root
				dominant_religion = owner.religion
			}
		}
	}
	immediate = {
		random_non_state_religion_location = {
			limit = {
				monthly_conversion > 0
				is_core_of = root
				NOT = {	has_location_modifier = missionary_failure	}
				any_neighbor_location = {
					owner ?= root
					dominant_religion = owner.religion
				}
			}
			save_scope_as = target_location
		}

	}

	option = {
		name = missionaries.2.a
		ai_chance = {	base = 0 }  #lets avoid rebels for now.
		change_societal_value = { type = mysticism_vs_jurisprudence value = societal_value_tiny_move_to_right }
		scope:target_location = {
			every_pop = {	add_pop_satisfaction = pop_satisfaction_severe_penalty	}
		}
	}
	option = {
		name = missionaries.2.b
		scope:target_location = { add_location_modifier = { modifier = missionary_failure months = 60 mode = add_and_extend }  }
		change_societal_value = { type = mysticism_vs_jurisprudence value = societal_value_tiny_move_to_left }
	}
}

missionaries.7 = {
	type = country_event
	title = missionaries.7.title
	desc = missionaries.7.desc
	trigger = {
		any_non_state_religion_location = {
			monthly_conversion > 0
			any_neighbor_location = {
				has_owner = yes
				owner ?= { at_war = yes }
			}
		}
		prestige > 10
		at_war = no
	}
	immediate = {
		random_non_state_religion_location = {
			limit = {
				monthly_conversion > 0
				any_neighbor_location = {
					owner ?= { at_war = yes }
				}
			}
			save_scope_as = target_location
		}

	}

	option = {
		name = missionaries.7.a
		add_gold = {
			value = scope:target_location.location_tax_base
			min = 20
			multiply = -2
		}
		change_societal_value = { type = mysticism_vs_jurisprudence value = societal_value_tiny_move_to_right }
	}
	option = {
		name = missionaries.7.b
		add_prestige = prestige_mild_penalty
		change_societal_value = { type = mysticism_vs_jurisprudence value = societal_value_tiny_move_to_right }
	}
}

missionaries.8 = {
	type = country_event
	title = missionaries.8.title
	desc = missionaries.8.desc
	trigger = {
		any_non_state_religion_location = {
			monthly_conversion > 0
		}
	}
	immediate = {
		ordered_non_state_religion_location = {
			limit = {	monthly_conversion > 0 }
			order_by = location_tax_base
			max = 1
			
			save_scope_as = target_location
		}
	
	}

	option = {
		name = missionaries.8.a

		ai_will_select = {
			value = 1
			if = {
				limit = {num_locations < 5 }
				subtract = 1
			}
		}
		add_gold = { value = scope:target_location.location_tax_base multiply = -4 max = -10 }
	}
	option = {
		name = missionaries.8.b

		ai_will_select = { value = 0 }

		add_government_power = government_power_weak_penalty
	}
}
