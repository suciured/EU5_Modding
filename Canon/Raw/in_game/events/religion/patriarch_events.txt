namespace = patriarch_events
patriarch_events.1 = {
	type = country_event
	title = patriarch_events.1.title
	desc = patriarch_events.1.desc
	trigger = {
		religion = religion:orthodox
		any_owned_location = {
			dominant_religion ?= religion:orthodox
			dominant_culture = owner.culture
			local_religious_unity < 1.0
			province = { NOT = { has_province_modifier = saints_enforced } }
		}
	}

	illustration_tags = {
		10 = angry
		10 = exterior
	}

	immediate = {
		hidden_effect = {
			random_owned_location = {
				limit = {
					dominant_religion ?= religion:orthodox
					dominant_culture = owner.culture
					local_religious_unity < 1.0
				}
				save_scope_as = local_saint_state_province
			}
		}
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
	}

	option = {
		name = patriarch_events.1.a
		scope:local_saint_state_province ?= {
			province ?= { 
				add_province_modifier = { modifier = saints_enforced years = 15 mode = add_and_extend }  
			}
		}
	}
	option = {
		name = patriarch_events.1.b
		scope:local_saint_state_province ?= {
			province ?= {
				every_location_in_province = {
					change_control = control_mild_penalty
				}
			}
		}
	}
}

patriarch_events.2 = {
	type = country_event
	title = patriarch_events.2.title
	desc = patriarch_events.2.desc
	trigger = {
		
		religion = religion:orthodox
		
		capital.area = {
			any_location_in_area = {
				dominant_religion ?= religion:orthodox
				has_building_with_at_least_one_level = cathedral
			}
		}
		has_ruler = yes
	}

	illustration_tags = {
		10 = angry
		10 = interior
	}

	immediate = {

		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		capital.area = {
			ordered_location_in_area = {
				order_by = development
				max = 1
				limit = {
					dominant_religion ?= religion:orthodox
					has_building_with_at_least_one_level = cathedral
				}
				save_scope_as = local_see_province
			}
		}
		ruler = { save_scope_as = target_character }
	}

	option = {
		name = patriarch_events.2.a
		add_religious_influence = religious_influence_mild_penalty
	}
	option = {
		name = patriarch_events.2.b
		change_gold_effect = { scale = -3.00 }
		add_religious_influence = religious_influence_mild_bonus
	}
}

patriarch_events.4 = {
	type = country_event
	title = patriarch_events.4.title
	desc = patriarch_events.4.desc
	trigger = {
		
		religion = religion:orthodox
		any_owned_location = {
			dominant_religion ?= religion:orthodox
			NOT = { province = owner.capital.province }
			dominant_culture = owner.culture
			province = { NOT = { has_province_modifier = rooting_out_heresy } }
		}
	}

	illustration_tags = {
		10 = armed
		10 = exterior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:burghers_estate }
		hidden_effect = {
			ordered_owned_location = {
				order_by = population
				max = 1
				limit = {
					dominant_religion ?= religion:orthodox
					NOT = { province = owner.capital.province }
					dominant_culture = owner.culture
					province = { NOT = { has_province_modifier = rooting_out_heresy } }
				}
				save_scope_as = judaizers_province
			}
		}
	}

	option = {
		name = patriarch_events.4.a
		scope:judaizers_province ?= {
			province ?= {
				add_province_modifier = { modifier = rooting_out_heresy years = 10 mode = add_and_extend }  
			}
		}
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_weak_bonus }
	}
	option = {
		name = patriarch_events.4.b
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_penalty }
	}
}

patriarch_events.5 = {
	type = country_event
	title = patriarch_events.5.title
	desc = patriarch_events.5.desc
	trigger = {
		
		religion = religion:orthodox
		any_owned_location = {
			dominant_religion ?= religion:orthodox
			has_building_with_at_least_one_level = orthodox_monastery
			NOT = {	has_variable = land_theft_from_monastics}
			NOT = { province = owner.capital.province }
		}
	}

	illustration_tags = {
		10 = armed
		10 = exterior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:clergy_estate }
		hidden_effect = {
			random_owned_location = {
				limit = {
					dominant_religion ?= religion:orthodox
					has_building_with_at_least_one_level = orthodox_monastery
					NOT = {	has_variable = land_theft_from_monastics}
					NOT = { province = owner.capital.province }
				}
				save_scope_as = monastic_province
			}
		}
	}

	option = {
		name = patriarch_events.5.a
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_penalty }
		scope:monastic_province = {
			change_development  = development_weak_bonus
			set_variable = { name = land_theft_from_monastics value = 1 }
		}
	}
	option = {
		name = patriarch_events.5.b
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_weak_bonus }
	}
}
