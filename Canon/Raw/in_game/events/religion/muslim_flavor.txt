muslim_flavor.4 = {
	type = country_event
	title = muslim_flavor.4.title
	desc = muslim_flavor.4.desc
	illustration_tags = {
		10 = regular
		10 = exterior
	}

	trigger = {
		religion.group = religion_group:muslim
		at_war = no
		has_regent = no
		gold >= {
			value = monthly_income_trade_and_tax
			multiply = 3
		}
		has_ruler = yes
		NOT = {
			ruler = { has_character_modifier = muslim_hajj }
		}
		country_has_estate = estate_type:clergy_estate
	}

	immediate = {
		ruler = { save_scope_as = target_ruler }
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
	}

	option = {
		name = muslim_flavor.4.a
		ruler = { add_character_modifier = { modifier = muslim_hajj months = -1 mode = add_and_extend } }
		change_gold_effect = { scale = -2.40 }
		trigger_event_silently = { id = muslim_flavor.5 days = 15 }

	}
	option = {
		name = muslim_flavor.4.b
		add_estate_satisfaction = { type = estate_type:clergy_estate
			value = estate_satisfaction_mild_penalty
		}
	}
}
