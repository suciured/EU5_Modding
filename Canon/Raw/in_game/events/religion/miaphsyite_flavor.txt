namespace = miaphysite_flavor

miaphysite_flavor.1 = {
	hide_portraits = yes
	type = location_event
	fire_only_once = yes
	title = miaphysite_flavor.1.title
	desc = miaphysite_flavor.1.desc
	
	trigger = {
		exists = scope:loser
		this = location:axum
		owner.religion = religion:miaphysite
		NOT = { scope:loser.religion = religion:miaphysite }
	}

	illustration_tags = {
		10 = happy
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = {
			foreground = estate_type:clergy_estate
			background = estate_type:clergy_estate
		}
	
		location:axum = {
			save_scope_as = target_location
		}
	}

	option = {
		name = miaphysite_flavor.1.a
		owner = {
			add_prestige = prestige_weak_bonus
			save_scope_as = from
			every_known_country = {
				limit = {
					is_ai = no
					religion = religion:miaphysite
				}
				trigger_event_silently = { id = miaphysite_flavor.2 }
			}
		}
	}
}

miaphysite_flavor.33 = {
	hide_portraits = yes
	type = country_event
	fire_only_once = yes
	title = miaphysite_flavor.33.title
	desc = miaphysite_flavor.33.desc
	trigger = {
		religion = religion:miaphysite
		NOT = {
			has_variable = reconnected_with_eastern_med_christians
		}
		capital = {
			continent = continent:africa
		}
		NOT = {
			any_owned_location = {
				continent = continent:europe
			}
		}
		NOT = {
			any_neighbor_country = {
				religion = religion:catholic
			}
		}
		NOT = {
			any_neighbor_country = {
				religion = religion:orthodox
			}
		}
		any_neighbor_country = {
			religion.group = religion_group:muslim
		}
	}

	illustration_tags = {
		10 = exterior
	}

	immediate = {
		event_illustration_estate_background_effect = { background = estate_type:clergy_estate }
		set_variable = { name = reconnected_with_eastern_med_christians value = 1 }
		save_scope_as = sent_diplomats
		random_neighbor_country = {
			limit = {
				religion.group = religion_group:muslim
			}
			save_scope_as = muslim_neighbor
		}
		random_country = {
			limit = {
				num_of_non_rural >= 6
				NOT = {
					is_at_war_with = root
				}
				OR = {
					religion = religion:catholic
					religion = religion:orthodox
				}
				capital = {
					OR = {
						region = region:france_region
						region = region:italy_region
						region = region:iberia_region
						region = region:balkan_region
						region = region:anatolia_region
					}
				}
			}
			save_scope_as = faraway_catholic
		}
	}

	option = {
		name = miaphysite_flavor.33.a
		custom_tooltip = miaphysite_flavor.33.a.tt
		change_gold_effect = { scale = -1.0 }
		save_scope_as = from
		scope:muslim_neighbor = { trigger_event_silently = { id = miaphysite_flavor.34 days = 60 } }
	}
	option = {
		name = miaphysite_flavor.33.b
		custom_tooltip = miaphysite_flavor.33.b.tt
		add_prestige = prestige_weak_penalty
	}
}
