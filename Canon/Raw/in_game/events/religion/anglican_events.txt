anglican_events.7 = {
	type = country_event
	title = anglican_events.7.title
	desc = anglican_events.7.desc
	
	trigger = {
		religion = religion:anglican
		any_cabinet_character = {
			religion = religion:anglican
		}
	}
	
	illustration_tags = {
        10 = angry
        10 = interior
    }
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		random_cabinet_character = {
			limit = {
				religion = religion:anglican
			}
			save_scope_as = minister
		}
	}
	
	option = {
		name = anglican_events.7.a
		scope:minister = {
			change_character_religion = religion:catholic
		}
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_extreme_penalty }
	}
	
	option = {
		name = anglican_events.7.b
		scope:minister = {
			change_character_religion = religion:catholic
		}
		
		remove_from_cabinet = scope:minister
		scope:minister = { add_character_modifier = { modifier = refuses_to_serve_in_cabinet years = -1 mode = add_and_extend }	}
	}
	
	option = {
		name = anglican_events.7.c
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
		kill_character = {
			target = scope:minister
			reason = beheading
			killer = ruler
		}
	}
}

anglican_events.10 = {
	hide_portraits = yes
	type = country_event
	title = anglican_events.10.title
	desc = anglican_events.10.desc
	trigger = {
		religion = religion:anglican
		current_age = age_5_absolutism
		has_colonial_charters = yes
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }
		random_colonial_charter = {
			save_scope_as = target_colony
			province_definition = {
				random_neighbor_province_definition = {
					limit = {
						any_location_in_province_definition = {
							is_land = yes
							has_owner = no
						}
						NOT = { root = { has_colonial_charter_in = prev } }
					}
					save_scope_as = neighbor_colony
				}
			}
		}
		
		random_province = {
			save_scope_as = origin
		}
	
	}

	option = {
		name = anglican_events.10.a
		create_colonial_charter = {
			target = scope:neighbor_colony
		}
		
		
		add_migration = {
			owner = this
			from = scope:origin.province_definition
			to = scope:neighbor_colony
			religion = religion:anglican
			type = pop_type:peasants
			amount = 0.1
			months = 60
		}
	}
	option = {
		name = anglican_events.10.b
		add_migration = {
			owner = this
			from = scope:origin.province_definition
			to = scope:target_colony.province_definition
			religion = religion:anglican
			type = pop_type:peasants
			amount = 0.05
			months = 60
		}
	}
}
