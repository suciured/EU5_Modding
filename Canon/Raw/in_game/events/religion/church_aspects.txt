church_aspects.1 = {
	hide_portraits = yes
	type = country_event
	title = church_aspects.1.title
	desc = church_aspects.1.desc

	trigger = {
		has_religious_aspect = religious_aspect:pacifism
		monthly_manpower > 0
		any_current_war = {
			is_an_attacker = root
		}
	}

	illustration_tags = {
		10 = armed
		10 = exterior
	}


	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:peasants background = pop_type:soldiers }
		random_country_at_war_with = {
			limit = {
				any_current_war = {
					is_an_attacker = root
				}
			}
			save_scope_as = pacifist_target
		}
	}

	option = {
		name = church_aspects.1.a

		add_war_exhaustion = war_exhaustion_weak_bonus
		add_manpower = { value = root.monthly_manpower multiply = -2 }
	}

	option = {
		name = church_aspects.1.b

		add_war_exhaustion = war_exhaustion_mild_penalty
	}
}

	}

		has_religious_aspect = religious_aspect:usury_allowed

				}
			}
		}

		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:clergy_estate }

		ruler = {
			save_scope_as = target_ruler
		}
		random_owned_non_rural_location = {
			limit = {
				NOT = {
					OR = {
						has_building_with_at_least_one_level = university
						has_building_with_at_least_one_level = cathedral
					}
				}
			}
			save_scope_as = target_location
		}
	}

	option = {
		name = church_aspects.7.a

		scope:target_location = {
			construct_building = {
				building_type = building_type:university
				cost_multiplier = 0
				cost_multiplier_reason = "academic_cheap_sponsorship"
			}
		}
	}

	option = {
		name = church_aspects.7.b

			construct_building = {
				building_type = building_type:cathedral
				cost_multiplier = 0

			}
		}
