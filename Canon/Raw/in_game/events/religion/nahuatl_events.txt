	hide_portraits = yes
	type = country_event
	title = nahuatl_events.2.title
	desc = nahuatl_events.2.desc

	trigger = {
		religion = religion:nahuatl
		NOT = {
			has_variable = nahuatl_warrior_born
		}
	}

	illustration_tags = {
		10 = happy
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		create_character = {
			estate = estate_type:peasants_estate
			culture = root.culture
			religion = root.religion

	}

	option = { #https://en.wikipedia.org/wiki/T%C4%93lpochcalli
		name = nahuatl_events.2.a
		trigger_event_silently = { id = nahuatl_events.3 years = 16 }

	}
	option = { #https://en.wikipedia.org/wiki/Calmecac

		add_government_power = government_power_extreme_penalty
