lutheranism_events.1 = {
	hide_portraits = yes
	type = country_event
	title = lutheranism_events.1.title
	desc = lutheranism_events.1.desc
	trigger = {
		religion = religion:lutheran
		capital.continent = continent:europe

		religion:catholic = {
			any_country_in_religion = {
				num_of_non_rural >= 5
				"religion_percentage_in_country(religion:lutheran)" > 0
				capital.continent = continent:europe
			}
		}
	}
	
	illustration_tags = {
		10 = interior
	}


	immediate = {
		event_illustration_estate_background_effect = { background = estate_type:clergy_estate }
		religion:catholic = {
			random_country_in_religion = {
				limit = {
					num_of_non_rural >= 5
					"religion_percentage_in_country(religion:lutheran)" > 0
					capital.continent = continent:europe
				}
				save_scope_as = reformer_fleeing_from_here

				create_character = {
					religion = religion:lutheran
					age = 33
					save_scope_as = reformer
					create_in_limbo = yes
				}
			}
		}
	}

	option = {
		name = lutheranism_events.1.a
		add_religious_influence = religious_influence_mild_bonus
		save_scope_as = from
		scope:reformer_fleeing_from_here = {
			add_opinion = { target = root modifier = protected_reformer }
			trigger_event_silently = { id = lutheranism_events.2 }
		}
		scope:reformer = { move_country = root }
		trigger_event_silently = { id = lutheranism_events.3 days = { 1000 2000 } }
	}
	option = {
		name = lutheranism_events.1.b
		add_religious_influence = religious_influence_mild_penalty
		scope:reformer_fleeing_from_here = {
			add_opinion = { target = root modifier = did_not_protect_reformer }
		}
		hidden_effect = { kill_character_silently = scope:reformer }
	}
}

		}

	option = {
		name = lutheranism_events.10.a
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
		scope:non_conformist = {
			every_location_in_province = {
				every_pop = {
					limit = {
						owner = root
						religion != religion:lutheran
					}
					split_pop = {
						fraction = 0.10
						religion = religion:lutheran
					}
					add_pop_satisfaction = pop_satisfaction_extreme_penalty
				}
			}
		}
	}
	option = {
		name = lutheranism_events.10.b

		scope:non_conformist = {

			}
		}
