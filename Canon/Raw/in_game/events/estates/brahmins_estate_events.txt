brahmins_estate_events.2 = {
	type = country_event
	title = brahmins_estate_events.2.title
	desc = brahmins_estate_events.2.desc

	illustration_tags = {
		10 = exterior
		10 = angry
	}

	immediate = {
		event_illustration_estate_in_location_effect = { estate_type = estate_type:clergy_estate }
	}

	trigger = {
		religion = religion:hindu
		"estate_power(estate_type:clergy_estate)" >= 0.30
		estate_satisfaction:clergy_estate < 0.50
		NOT = {	has_country_modifier = religious_court_life	}
	}

	option = {
		name = brahmins_estate_events.2.a
		add_country_modifier = { modifier = religious_court_life years = 15 mode = add_and_extend }  
	}
	option = {
		name = brahmins_estate_events.2.b
		add_stability = stability_extreme_penalty
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_extreme_penalty }
	}
}

brahmins_estate_events.10 = {
	type = country_event
	title = brahmins_estate_events.10.title
	desc = brahmins_estate_events.10.desc
	trigger = {
		religion = religion:hindu
		estate_satisfaction:nobles_estate < 0.30
		estate_satisfaction:clergy_estate >= 0.30
		any_owned_location = {
			dominant_religion = religion:hindu
			modifier:fort_level >= 0
			is_capital = no
		}
	}

	illustration_tags = {
		10 = exterior
		10 = regular
	}


	immediate = {
		if = {
			limit = {
				any_owned_location = {
					dominant_religion = religion:hindu
					modifier:fort_level >= 0
					is_capital = no
				}
			}
			random_owned_location = {
				limit = {
					dominant_religion = religion:hindu
					modifier:fort_level >= 0
					is_capital = no
				}
				save_scope_as = brahmin_fort_province
			}
		}
		event_illustration_estate_in_location_effect = { estate_type = estate_type:clergy_estate }
	}

	option = {
		name = brahmins_estate_events.10.a
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_penalty }
	}
	option = {
		name = brahmins_estate_events.10.b
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_penalty }
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_bonus }
	}
}
