clergy_estate_events.1 = {
	type = country_event
	title = clergy_estate_events.1.title
	desc = clergy_estate_events.1.desc
	trigger = {
		religion = religion:catholic
		"estate_power(estate_type:clergy_estate)" >= 0.10
		num_loans > 0
	}

	illustration_tags = {
		10 = angry
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
	}

	option = {
		name = clergy_estate_events.1.a
		add_gold = { value = root.estate_tax_base:clergy_estate	multiply = 10 }
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_extreme_penalty	}
		add_gold_to_estate = {
			estate_type = estate_type:clergy_estate
			value = { value = root.estate_tax_base:clergy_estate multiply = -10 	}
		}
	}
	option = {
		name = clergy_estate_events.1.b
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus	}
	}
}
