namespace = exploration_success

# distinguished midshipman
exploration_success.1 = {
	type = country_event

	title = exploration_success.1.title
	desc = exploration_success.1.desc

	trigger = {
		any_core_location = { is_port = yes }
	}

	immediate = {
		random_core_location = {
			limit = { is_port = yes }
			save_scope_as = midshipman_location
			random_pop = {
				save_scope_as = midshipman_pop
			}
		}
	}

	option = {
		name = exploration_success.1.a

		create_character = {
			culture = scope:midshipman_pop.culture
			religion = scope:midshipman_pop.religion
			age = 18
			female = no
			birth_location = scope:midshipman_location
			trait_category = explorer
		}
		change_gold_effect = { scale = -1 }
	}

	option = {
		name = exploration_success.1.b
		add_navy_tradition = navy_tradition_mild_bonus
	}
}
