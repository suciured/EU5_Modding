# explorer marries a native
exploration_monthly.11 = {
	type = exploration_event

	title = exploration_monthly.11.title
	desc = exploration_monthly.11.desc




	trigger = {
		leader ?= {
			is_married = no
			is_female = no
			OR = {
				has_dynasty = no
				AND = {
					has_dynasty = yes
					owner = {
						ruler ?= {
							dynasty ?= {
								this != ROOT.leader.dynasty
							}
						}
					}
				}
			}
			NOT = {
				has_estate = estate_type:clergy_estate
			}
		}
		location = {
			population > 0.100
			any_pop = { pop_type = pop_type:peasants }
		}
	}

	immediate = {
		location = {
			random_pop = {
				limit = { pop_type = pop_type:peasants }
				save_scope_as = bride_pop
			}
			save_scope_as = marriage_location
		}
		owner = {
			create_character = {
				culture = scope:bride_pop.culture
				religion = scope:bride_pop.religion
				age = 18
				female = yes
				birth_location = ROOT.location
				save_scope_as = bride
				estate = estate_type:peasants_estate
			}
		}
	}

	option = {
		name = exploration_monthly.11.a

		trigger = { NOT = { scope:bride.religion = ROOT.owner.religion } }

		scope:bride = {
			marry_character = ROOT.leader
			change_character_religion = ROOT.owner.religion
		}
		change_exploration_progress = {
			value = exploration_time
			multiply = 0.05
		}
	}

	option = {
		name = exploration_monthly.11.b

		scope:bride = { marry_character = ROOT.leader }

		change_exploration_progress = {
			value = exploration_time
			multiply = 0.1
		}
	}

	option = {
		name = exploration_monthly.11.c

		hidden_effect = { kill_character_silently = scope:bride }
	}
}

	desc = exploration_monthly.21.desc


	trigger = {

	immediate = {

			create_character = {
