cultural_event.13 = {
	hide_portraits = yes
	type = country_event
	title = cultural_event.13.title
	desc = cultural_event.13.desc
	illustration_tags = {
		10 = happy
		10 = interior
	} 
	trigger = {
		exists = last_valid_ruler
		has_regent = yes
		government_type = government_type:monarchy
		any_artist = {
			artist_type = writer
		}
	}
	illustration_tags = {
 		10 = exterior
		10 = happy
 	}

	immediate = {
		random_artist = {
			limit = {
				artist_type = writer
			}
			save_scope_as = target_artist
			
		}
		last_valid_ruler = {
			save_scope_as = target_character
		}
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:nobles_estate }
	}

	option = {
		name = cultural_event.13.a
		capital = {
			create_art = {
				artist = scope:target_artist
				target = scope:target_character
				quality = 66
				type = work_of_art_type:poem
				key = national_epic
			}
		}
		change_gold_effect = { scale = -1 }
	}
	option = {
		name = cultural_event.13.b
		add_prestige = prestige_weak_penalty
	}
}

cultural_event.14 = { # Sponsoring University
	type = country_event
	title = cultural_event.14.t
	desc = cultural_event.14.desc
	image = "gfx/interface/illustrations/institutions/scientific_revolution.dds"
	trigger = {
		has_embraced_institution = institution:renaissance
		any_owned_location = {
			has_building_with_at_least_one_level = university
		}
	}



	immediate = {
		random_owned_location = {
			limit = { has_building_with_at_least_one_level = university }
			save_scope_as = target_location
		}
		create_character = {
			estate = estate_type:burghers_estate
			culture = scope:target_location.dominant_culture
			religion = scope:target_location.dominant_religion
			add_trait = trait:industrious
			add_trait = trait:scholar
			age = 37
			birth_location = scope:target_location
			create_in_limbo = yes
			save_scope_as = target_character_scope
		}
	}

	option = { # Gain skillful advisor
		name = cultural_event.14.a

		change_gold_effect = { scale = -6 }
		add_research_progress = research_progress_mild_bonus
		scope:target_character_scope = { move_country = root }
	}

	option = { # Refuse to give nerds money
		name = cultural_event.14.b
		add_prestige = prestige_weak_penalty
		kill_character_silently = scope:target_character_scope
		scope:target_location = {
			every_pop = {
				limit = {
					owner = root
					pop_type = pop_type:burghers
				}
				add_pop_satisfaction = pop_satisfaction_severe_penalty
			}
		}
	}
}
