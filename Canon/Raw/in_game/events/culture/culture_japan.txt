culture_japan.1 = { #O-Ie Sōdō
	type = country_event
	title = culture_japan.1.title
	desc = culture_japan.1.desc
	trigger = {
		current_age = age_5_absolutism
		government_type = government_type:monarchy
		culture = { has_culture_group = culture_group:japanese_group }
		has_heir = yes
		heir ?= {
			is_female = no
			is_adult = yes
		}
		legitimacy < 50
		prestige < 40
	}

	illustration_tags = {
        10 = angry
        10 = interior
    }
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		heir ?= {
			save_scope_as = target_character
		}
	}

	option = {
		name = culture_japan.1.a
		kill_character = scope:target_character
	}
	option = {
		name = culture_japan.1.b
		add_legitimacy = legitimacy_mild_penalty
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_extreme_penalty }
	}
}

culture_japan.3 = { #Heirless
	type = country_event
	title = culture_japan.3.title
	desc = culture_japan.3.desc
	
	trigger = {
		culture = { has_culture_group = culture_group:japanese_group }
		has_ruler_in_country = yes
		ruler ?= {
			NOT = { any_child = { is_alive = yes } }
		}
		has_heir = no
		any_character = {
			has_dynasty = no
			NOT = { exists = father }
			is_female = no
			age_in_years < {
				value = root.ruler.age_in_years
				subtract = 10
			}
			is_artist = no
		}
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		random_character = {
			limit = {
				has_dynasty = no
				NOT = { exists = father }
				is_female = no
				age_in_years < {
					value = root.ruler.age_in_years
					subtract = 10
				}
				is_artist = no
			}
			save_scope_as = target_character
		}
		ruler ?= { save_scope_as = target_ruler }
	}

	option = {
		name = culture_japan.3.a
		add_prestige = prestige_weak_penalty
	}
	option = {
		name = culture_japan.3.b
		
		if  = {
			limit = {
				#Make sure that the new "parent" is not a descendant of the new "kid"
				scope:new_ruler ?= {
					NOT = {
						any_descendant = {
							this = scope:old_ruler 
						} 
					}
				}
			}
			scope:target_ruler = {
				adopt_character = scope:target_character
			}
		}
		scope:target_character = {
			if = {
				limit = {
					NOT = { has_estate = estate_type:nobles_estate }
				}
				change_character_estate = estate_type:nobles_estate
			}
		}
		add_legitimacy = legitimacy_weak_penalty
	}
}

culture_japan.1200 = { #Declining Power of the Shogunate
	type = country_event
	title = culture_japan.1200.title
	desc = culture_japan.1200.desc

	trigger = {
		NOT = { is_situation_active = situation:nanbokuchou }
		NOT = { is_situation_active = situation:sengoku }
		culture = { has_culture_group = culture_group:japanese_group }
		exists = international_organization:japanese_shogunate
		is_leader_of_international_organization = international_organization:japanese_shogunate
		OR = {
			legitimacy <= 60
			stability <= 50
			war_exhaustion >= 5
		}
		any_owned_location = {
			NOT = { has_building = building_type:yamashiro }
		}
		international_organization:japanese_shogunate = {
			any_international_organization_member = {
				country_type = building
				is_japanese_emperor_or_shogun = no
				NOT = { is_at_war_with = root }
				is_subject = no
				opinion = { target = root value >= 50 }
			}
		}
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		save_scope_as = target_shogunate
		international_organization:japanese_shogunate = {
			random_international_organization_member = {
				limit = {
					country_type = building
					is_japanese_emperor_or_shogun = no
					NOT = { is_at_war_with = root }
					is_subject = no
					opinion = { target = root value >= 50 }
				}
				save_scope_as = target_clan_1
			}
			random_international_organization_member = {
				limit = {
					country_type = building
					is_japanese_emperor_or_shogun = no
					NOT = { is_at_war_with = root }
					is_subject = no
					this != scope:target_clan_1
					opinion = { target = root value >= 50 }
				}
				save_scope_as = target_clan_2
			}
			random_international_organization_member = {
				limit = {
					country_type = building
					is_japanese_emperor_or_shogun = no
					NOT = { is_at_war_with = root }
					is_subject = no
					this != scope:target_clan_1
					this != scope:target_clan_2
					opinion = { target = root value >= 50 }
				}
				save_scope_as = target_clan_3
			}
		}
	}

	option = {
		name = culture_japan.1200.a
		scope:target_clan_1 = {
			trigger_event_non_silently = culture_japan.1201
		}
		if = {
			limit = { religion = { has_honor = yes } }
			add_honor = honor_weak_penalty
		}
	}

	option = {
		name = culture_japan.1200.b
		scope:target_clan_2 = {
			trigger_event_non_silently = culture_japan.1201
		}
		if = {
			limit = { religion = { has_honor = yes } }
			add_honor = honor_weak_penalty
		}
	}

	option = {
		name = culture_japan.1200.c
		scope:target_clan_3 = {
			trigger_event_non_silently = culture_japan.1201
		}
		if = {
			limit = { religion = { has_honor = yes } }
			add_honor = honor_weak_penalty
		}
	}

	option = {
		name = culture_japan.1200.d
		add_prestige = prestige_mild_bonus
	}
}
