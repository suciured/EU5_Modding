namespace = tribe_flavor

tribe_flavor.3 = {
	type = country_event
	title = tribe_flavor.3.title
	desc = tribe_flavor.3.desc
	trigger = {
		country_type = pop
		government_type = government_type:tribe
		at_war = yes
		NOT = {	has_country_modifier = defensive_stance	}
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
	}
	
	option = {
		name = tribe_flavor.3.a
		add_country_modifier = { modifier = defensive_stance years = 10 mode = add_and_extend }
	}
}

tribe_flavor.8 = {
	hide_portraits = yes
	type = country_event
	title = tribe_flavor.8.title
	desc = tribe_flavor.8.desc
	trigger = {
		country_type = pop
		government_type = government_type:tribe

		any_owned_nomad_pop = {
			count > 0
		}
	}

	illustration_tags = {
        10 = angry
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		random_owned_nomad_pop = {
			location = {
				save_scope_as = target
			}
		}
	}

	option = {
		name = tribe_flavor.8.a
		scope:target = {
			change_prosperity  = -0.25
		}
	}
}

tribe_flavor.9 = {
	hide_portraits = yes
	type = country_event
	title = tribe_flavor.9.title
	desc = tribe_flavor.9.desc
	trigger = {
		country_type = pop
		government_type = government_type:tribe
		at_war = no

		any_other_country = {
			capital.area = root.capital.area
		}
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		random_other_country = {
			limit = { 	capital.area = root.capital.area }
			save_scope_as = target
		}
	}

	option = {
		name = tribe_flavor.9.a
		scope:target = {
			random_list = {
				50 = {
					add_opinion = { target = root modifier = dancing_on_the_knife_good }
				}
				50 = {
					add_opinion = { target = root modifier = dancing_on_the_knife_bad }
				}
			}
		}
	}
}
