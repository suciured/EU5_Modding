#Unpopularity of the Doge - Nobles / Burghers oust him
republics.34 = {
	hide_portraits = yes
	type = country_event
	title = republics.34.title
	desc = republics.34.desc
	trigger = {
		government_type = government_type:republic
		succession_law = heir_selection:doge_election
		estate_satisfaction:nobles_estate < 0.3
		estate_satisfaction:burghers_estate < 0.3
		ruler = {
			NOT = {
				has_variable = was_tried_to_be_ousted
			}
		}
	}

	immediate = {
		create_rebel = {
			category = pretender
			name = upset_nobles_pretender
			save_scope_as = upset_nobles_pretender
		}
		ruler = {
			set_variable = {
				name = was_tried_to_be_ousted
				value = 1
			}
		}
		create_character = {
			estate = estate_type:nobles_estate
			birth_location = root.capital
			save_scope_as = nobles_pretender
		}
	}

	option = {
		name = republics.34.a
		add_stability = stability_mild_penalty
		add_estate_satisfaction = { type = estate_type:nobles_estate
			value = estate_satisfaction_extreme_bonus
		}
		add_estate_satisfaction = { type = estate_type:burghers_estate
			value = estate_satisfaction_extreme_bonus
		}
		set_new_ruler = scope:nobles_pretender
		ai_chance = {
			factor = 95
			modifier = {
				factor = 0
				ruler = root.ruler
			}
		}
	}
	option = {
		name = republics.34.b

		scope:nobles_pretender = {
			change_character_allegiance = scope:upset_nobles_pretender
		}
		custom_tooltip = {
			text = upset_nobles_pretender_tt
			every_owned_location = {
				limit = {
					area = root.capital.area
				}
				every_pop = {
					limit = {
						owner = root
						OR = {
							pop_type = pop_type:nobles
							pop_type = pop_type:burghers
						}
					}
					add_pop_satisfaction = pop_satisfaction_ultimate_penalty
					change_pop_allegiance = scope:upset_nobles_pretender
				}
			}

		}
		ai_chance = {
			factor = 5
		}
	}
}
