laws.0001 = { # State-critical Art
	type = country_event

	title = laws.0001.t
	desc = laws.0001.desc

	image = "gfx/interface/illustrations/institutions/renaissance.dds"

	trigger = {
		has_policy = no_censorship
		any_artist = {
			OR = {
				artist_type = philosopher
				artist_type = jurist
				artist_type = scientist
			}
		}
	}

	immediate = {
		random_artist = {
			limit = {
				OR = {
					artist_type = philosopher
					artist_type = jurist
					artist_type = scientist
				}
			}
			save_scope_as = critical_artist
		}
		capital = {
			create_art = {
				type = work_of_art_type:treatise
				artist = scope:critical_artist
				location = scope:critical_artist.location
				quality = 10
				save_scope_as = critical_art
			}
		}
	}

	option = { # Ok, whatever
		name = laws.0001.a

		add_government_power = government_power_mild_penalty
	}

	option = { # I'm going to take that painting and shove it u-
		name = laws.0001.b

		add_stability = stability_weak_penalty

		kill_character = {
			target = scope:critical_artist
			location = scope:critical_artist.location
			reason = execution
		}
		destroy_art = scope:critical_art
	}
}
