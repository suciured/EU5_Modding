	}

		10 = happy
		10 = interior
	}

	immediate = {

		random_owned_location = {
			create_dynasty_from_location = random

			limit = {

			}

				dynasty = scope:target_dynasty

			}
		}
		else = {
			create_character = {
				estate = estate_type:nobles_estate
				save_scope_as = target_character
			}
		}
	}

	option = {
		name = societal_values.104.a
		add_prestige = prestige_severe_bonus
	}
}

societal_values.105 = {
	type = country_event
	title = societal_values.105.title
	desc = societal_values.105.desc
	trigger = {
		societal_value:serfdom_vs_free_subjects < -20
		NOT = { has_estate_privilege = estate_privilege:peasants_free_peasantry }
		any_owned_rural_location = { count >= 1 }
	}

	illustration_tags = {
		10 = angry
		10 = exterior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		ordered_owned_rural_location = {
			order_by = population
			limit = {
				is_capital = no
			}
			max = {
				value = 1
				add = {
					value = root.num_provinces
					multiply = 0.2
				}
			}
			check_range_bounds = no
			set_variable = revolt_location_variable
		}
		random_owned_rural_location = {
			limit = {
				has_variable = revolt_location_variable
			}
			save_scope_as = target_location
		}
	}

	option = {
		name = societal_values.105.a

		custom_tooltip = societal_values.105.a.tt
		hidden_effect = {
			every_owned_rural_location = {
				limit = { has_variable = revolt_location_variable }
				every_pop = {
					limit = {
						owner = root
						pop_type = pop_type:peasants
					}
					add_pop_satisfaction = pop_satisfaction_ultimate_penalty
				}
			}
		}
	}
}
