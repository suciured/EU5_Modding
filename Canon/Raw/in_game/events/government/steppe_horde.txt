steppe_horde.2 = {
	type = country_event
	title = steppe_horde.2.title
	desc = steppe_horde.2.desc
	trigger = {
		country_type = army
		government_type = government_type:steppe_horde

		any_owned_location = {
			is_location_dominant_religion_pagan = yes
			NOT = {	has_location_modifier = ancestral_temple	}
		}

	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		random_owned_location = {
			limit = {
				is_location_dominant_religion_pagan = yes
			}
			save_scope_as = target_location
			dominant_religion = { save_scope_as = target_religion }
		}
	}

	option = {
		name = steppe_horde.2.a
		scope:target_location = { add_location_modifier = { modifier = ancestral_temple years = 10 mode = add_and_extend } }
		add_prestige = prestige_weak_bonus
	}

	option = {
		name = steppe_horde.2.b
		scope:target_location = {
			every_pop = {
				limit = {
					owner = root
					religion = scope:target_religion
				}
				add_pop_satisfaction = pop_satisfaction_severe_penalty
			}
		}
		add_stability = stability_weak_bonus
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
	}
}

steppe_horde.3 = {
	type = country_event
	title = steppe_horde.3.title
	desc = steppe_horde.3.desc
	trigger = {
		country_type = army
		government_type = government_type:steppe_horde

		at_war = yes
		NOT = {
			has_country_modifier = blue_dragon
		}
		NOT = {
			has_country_modifier = white_tiger
		}
		has_ruler = yes
		any_owned_location = {
			is_location_dominant_religion_pagan = yes
		}
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
		random_owned_location = {
			limit = {
				is_location_dominant_religion_pagan = yes
			 }
			save_scope_as = target_location
			dominant_religion ?= { save_scope_as = target_religion }
			dominant_culture ?= { save_scope_as = target_culture }
		}
		create_character = {
			birth_location = scope:target_location
			estate = estate_type:clergy_estate
			save_scope_as = target_character
			create_in_limbo = yes
		}
		scope:target_character = {
			if = {
				limit = { exists = scope:target_religion }
				change_character_culture = scope:target_religion
			}
			if = {
				limit = { exists = scope:target_culture }
				change_character_religion = scope:target_culture
			}
		}
		ruler = { save_scope_as = target_ruler }
	}

	option = {
		name = steppe_horde.3.a
		scope:target_character = { move_country = root }
		add_country_modifier = { modifier = blue_dragon years = 10 mode = add_and_extend }
	}

	option = {
		name = steppe_horde.3.b
		kill_character = {
			target = scope:target_character
			killer = scope:target_ruler
			reason = execution
		}
		add_stability = stability_mild_bonus
	}
}
