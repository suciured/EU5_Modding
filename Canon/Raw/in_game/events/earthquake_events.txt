scripted_effect minor_earthquake_payment = {
	add_gold = {
		value = -25
		add = {
			value = scope:target_location.location_tax_base
			multiply = -7
		}
	}
}

scripted_effect major_earthquake_payment = {
	add_gold = {
		value = -50
		add = {
			value = scope:target_location.location_tax_base
			multiply = -15
		}
	}
}


scripted_effect catastrophic_earthquake_payment = {
	add_gold = {
		value = -100
		add = {
			value = scope:target_location.location_tax_base
			multiply = -20
		}
	}
}

scripted_effect minor_earthquake_damage = {
	change_prosperity = prosperity_severe_penalty
	every_pop = {
		add_pop_size = {
			value = pop_size
			multiply = -0.05
		}
	}
	ordered_buildings_in_location = {
		order_by = building_level
		max = {
			value = 1
			add = {
				value = num_buildings
				divide = 15
				ceiling = yes
			}
		}
		location = { minor_destroy_building_levels_in_location = yes }
	}
	every_unit_in_location = {
		damage_unit_percent = { 0.33 1 }
		damage_unit_morale_percent = { 0.5 1 }
	}
	every_neighbor_location = { 
		change_prosperity = prosperity_weak_penalty
		every_pop = {
			add_pop_size = {
				value = pop_size
				multiply = -0.05
			}
		}
		ordered_buildings_in_location = {
			order_by = building_level
			max = {
				value = 1
				add = {
					value = num_buildings
					divide = 20
				}
			}
			location = { minor_destroy_building_levels_in_location = yes }
		}
		every_unit_in_location = {
			damage_unit_percent = { 0 0.33 }
			damage_unit_morale_percent = { 0.2 0.5 }
		}
	}
}

scripted_effect catastrophic_earthquake_damage = {
	change_prosperity = prosperity_radical_penalty
	every_pop = {
		add_pop_size = {
			value = pop_size
			multiply = -0.2
		}
	}
	ordered_buildings_in_location = {
		order_by = building_level
		max = {
			value = 1
			add = {
				value = num_buildings
				divide = 8
				ceiling = yes
			}
		}
		location = { catastrophic_destroy_building_levels_in_location = yes }
	}
	every_unit_in_location = {
		destroy_unit = yes
	}
	every_neighbor_location = {
		change_prosperity = prosperity_extreme_penalty
		every_pop = {
			add_pop_size = {
				value = pop_size
				multiply = -0.1
			}
		}
		ordered_buildings_in_location = {
			order_by = building_level
			max = {
				value = 1
				add = {
					value = num_buildings
					divide = 10
				}
			}
			location = { major_destroy_building_levels_in_location = yes }
		}
		every_unit_in_location = {
			damage_unit_percent = { 0.5 1 }
			damage_unit_morale_percent = { 0.75 1 }
		}
	}
}

earthquake_events.1 = { #Earthquake in Constantinople
	type = country_event
	title = earthquake_events.1.title
	desc = earthquake_events.1.desc
	fire_only_once = yes
	trigger = {
		owns = location:constantinople
		current_year >= 1338.6.1
		current_year <= 1365.6.10
	}
	image = "gfx/interface/illustrations/event/backgrounds/special/earthquake.dds"

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		location:constantinople = { save_scope_as = target_location }
	}

	option = {
		name = earthquake_events.1.a
		historical_option = yes
		location:constantinople = {
			change_prosperity = prosperity_severe_penalty
		}
		add_government_power = government_power_extreme_penalty
		add_religious_influence_if_valid = { VALUE = religious_influence_mild_penalty }
	}

	option = {
		name = earthquake_events.1.b
		major_earthquake_payment = yes
		location:constantinople = {
			change_prosperity = prosperity_weak_penalty
		}
	}
}

 earthquake_events.2 = { #Crimea
 	type = country_event
 	title = earthquake_events.title
 	desc = earthquake_events_minor.desc
 	fire_only_once = yes

 	trigger = {
 		owns = location:lusta
 		current_year >= 1338
 		current_year <= 1346
 	}

 	image = "gfx/interface/illustrations/event/backgrounds/special/earthquake.dds"
	
 	immediate = {
 		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }

 		location:lusta = { save_scope_as = target_location }
 	}
		
 	option = {
 		name = earthquake_events_minor.a

 		custom_tooltip = earthquake_events_minor.a.tt
 		hidden_effect = {
 			scope:target_location = {
 				minor_earthquake_damage = yes
 			}
 		}
 	}
	
 	option = {
 		name = earthquake_events_minor.b

 		minor_earthquake_payment = yes
 		custom_tooltip = earthquake_events_minor.b.tt
 		hidden_effect = {
 			scope:target_location = {
 				minor_earthquake_damage = yes
			}
		}
		scope:target_location = {
			add_location_modifier = {
				modifier = minor_recovery_effort
				years = 10
				mode = add_and_extend
			}

		}
 	}
 }

earthquake_events.1000 = { #The Gallipoli Earthquake - Be mindful - connected to earthquake_events.1001! 
	hide_portraits = yes
    type = country_event
	title = earthquake_events.1000.title
	desc = earthquake_events.1000.desc

	fire_only_once = yes

	historical_info = earthquake_events.1000.historical_info

	trigger = {
		owns = location:gallipoli
		#Ten year margin
		current_date < 1364.3.1
		current_date > 1354.3.1
	}

	image = "gfx/interface/illustrations/event/backgrounds/special/earthquake.dds"
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }

		location:gallipoli = {
			save_scope_as = target_location
		}

		#We try and get the Turkish Nemesis of the Player
		if  = {
			limit = {
				any_neighbor_country = {
					earthquake_is_turkish_nemesis = yes
				}
			}
			ordered_neighbor_country = {
				limit = {
					earthquake_is_turkish_nemesis = yes
				}
				order_by = total_population
				max = 1
				
				save_scope_as = turkish_nemesis
			}
		}
		#If there was no Turkic rival we try and get another Turkish nemesis
		else_if = {
			limit = {
				any_neighbor_country = {
					earthquake_is_turkish_new_nemesis = yes
				}
			}
			ordered_neighbor_country = {
				limit = {
					earthquake_is_turkish_new_nemesis = yes
				}
				order_by = total_population
				max = 1
				
				save_scope_as = turkish_nemesis
			}
		}
	}
		
	option = {
		name = earthquake_events.1000.a
        historical_option = yes
		
		custom_tooltip = flavor_nap.101.a.tt #Same text

		scope:target_location = {
			change_prosperity = prosperity_ultimate_penalty
		}
		hidden_effect = {
			scope:target_location = {
				#Kill pops
				every_pop = {
					add_pop_size = {
						value = pop_size
						multiply = -0.30
					}
				}
				#Destroy military Buildings
				destroy_building = "building(building_type:stockade|owner)"
				destroy_building = "building(building_type:castle|owner)"
				destroy_building = "building(building_type:bastion|owner)"
				destroy_building = "building(building_type:star_fort|owner)"
				destroy_building = "building(building_type:fortress|owner)"
				
				#Destroy Half the other buildings
				ordered_buildings_in_location = {
					order_by = building_level
					max = {
						value = 1
	
						add = {
							value = num_buildings
							divide = 2
						}
					}
					check_range_bounds = no
					location = { catastrophic_destroy_building_levels_in_location = yes }
				}
				#Affect neighboor locations
				every_neighbor_location = { 
					limit = {
						is_port = yes
					}
					change_prosperity = prosperity_severe_penalty
					every_pop = {
						add_pop_size = {
							value = pop_size
							multiply = -0.05
						}
					}
				} 
			}
			if = {
				limit = {
					exists = scope:turkish_nemesis
				}
				scope:turkish_nemesis = { 
					trigger_event_non_silently = earthquake_events.1001
				}
			}	
		}
	}
	
	option = {
		name = earthquake_events.1000.b
		
		custom_tooltip = flavor_nap.101.a.tt #Same text

		major_earthquake_payment = yes

		scope:target_location = {
			change_prosperity = prosperity_severe_penalty
		}
		hidden_effect = {
			scope:target_location = {
				#Kill pops
				every_pop = {
					add_pop_size = {
						value = pop_size
						multiply = -0.20
					}
				}
				#Destroy military Buildings
				destroy_building = "building(building_type:stockade|owner)"
				destroy_building = "building(building_type:castle|owner)"
				destroy_building = "building(building_type:bastion|owner)"
				destroy_building = "building(building_type:star_fort|owner)"
				destroy_building = "building(building_type:fortress|owner)"
				
				#Destroy a sixth of the other buildings
				ordered_buildings_in_location = {
					order_by = building_level
					max = {
						value = 1
	
						add = {
							value = num_buildings
							divide = 6
						}
					}
				
					location = { major_destroy_building_levels_in_location = yes }
				}
				#Affect neighboor locations
				every_neighbor_location = { 
					limit = {
						is_port = yes
					}
					change_prosperity = prosperity_mild_penalty
					every_pop = {
						add_pop_size = {
							value = pop_size
							multiply = -0.05
						}
					}
				} 
			}
			if = {
				limit = {
					exists = scope:turkish_nemesis
				}
				scope:turkish_nemesis = { 
					trigger_event_non_silently = earthquake_events.1001
				}
			}
		}	
	}
}
