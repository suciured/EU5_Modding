prices.1 = {
	type = country_event
	fire_only_once = yes
	title = prices.1.title
	desc = prices.1.desc
	trigger = {
		capital.market ?= {
			location.continent = continent:europe
			has_merchant = root
			most_powerful_merchant = root
		}
		any_market_with_merchants = {
			location.region = region:guinea_region
			has_merchant = root
			most_powerful_merchant = root
		}
		any_market_with_merchants = {
			location.continent = continent:america
			has_merchant = root
			most_powerful_merchant = root
		}
	}
	
	
	image = "gfx/interface/icons/trade_goods/illustrations/icon_goods_slaves_goods.dds"

	option = {
		name = prices.1.a
		add_prestige = prestige_extreme_bonus

		every_market_with_merchants = {
			limit = {
				location.continent = continent:europe
			}
			add_temporary_demand = { type = demand:triangle_trade_europe months = 240 }
		}

		every_market_with_merchants = {
			limit = {
				location.continent = continent:america
			}
			add_temporary_demand = { type = demand:triangle_trade_america months = 240 }
		}

		every_market_with_merchants = {
			limit = {
				location.region = region:guinea_region
			}
			add_temporary_demand = { type = demand:triangle_trade_africa months = 240 }

	}
}
