namespace = mercantilism_events
mercantilism_events.1 = {
	type = country_event
	title = mercantilism_events.1.title
	desc = mercantilism_events.1.desc
	trigger = {
		current_age_or_later = { age = age_4_reformation }
		societal_value:mercantilism_vs_free_trade < 20
		any_province = {
			is_starving = yes
			region = root.capital.region
			NOT = {	has_province_modifier = refused_food_import }
		}
	}

	illustration_tags = {
        10 = angry
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		random_province = {
			limit = {
				is_starving = yes
				region = root.capital.region
				NOT = {	has_province_modifier = refused_food_import }
			}
			save_scope_as = starving_province
		}

	}

	option = {
		name = mercantilism_events.1.a
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_minor_move_to_right
		}
	}
	option = {
		name = mercantilism_events.1.b
		scope:starving_province ?= {
			add_province_modifier = { modifier = refused_food_import months = 60 mode = add_and_extend }  
		}
		
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_tiny_move_to_left
		}
	}
}

mercantilism_events.2 = {
	hide_portraits = yes
	type = country_event
	title = mercantilism_events.2.title
	desc = mercantilism_events.2.desc
	trigger = {
		current_age_or_later = { age = age_4_reformation }
		NOT = {
			any_subject = {
				is_subject_type = colonial_nation
				capital = {
					continent = continent:america
				}
			}
		}
		capital  = { 
			NOT = { continent = continent:america }
			market ?= {
				any_import = {
					NOT = { owner ?= root }
					from_market ?= {
						location = {
							continent = continent:america
						}
					}
				}
			}
		}
	}

	illustration_tags = {
        10 = angry
        10 = exterior
    }

	immediate = {		
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }
		capital = {
			market = {
				random_import = {
					limit = {
						NOT = { owner = root }
						from_market = {
							location = {
								continent = continent:america
							}
						}
					}
					owner = { save_scope_as = american_importer }
				}
			}
		}
	}

	option = {
		name = mercantilism_events.2.a
		
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_minor_move_to_right
		}
		
		scope:american_importer = {
			add_opinion = { target = root modifier = opinion_pleased }
		}
	}
	option = {
		name = mercantilism_events.2.b
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_tiny_move_to_left
		}
		scope:american_importer = {
			add_opinion = { target = root modifier = opinion_insulted }
		}
	}
}

mercantilism_events.3 = {
	type = country_event
	title = mercantilism_events.3.title
	desc = mercantilism_events.3.desc
	
	image = "gfx/interface/icons/trade_goods/illustrations/icon_goods_iron.dds"
	
	trigger = {
		army_size > 0
		exists = capital
		capital.market ?= {
			NOT = { is_produced_in_market = goods:iron } 
			NOT = { is_traded_in_market = goods:iron } 
		}
	}

	option = {
		name = mercantilism_events.3.a
		add_army_tradition = army_tradition_weak_bonus
	}
	option = {
		name = mercantilism_events.3.b
		if = {
			limit = {
				current_age_or_later = { age = age_4_reformation }
			}
			change_societal_value = {
				type = mercantilism_vs_free_trade
				value = societal_value_tiny_move_to_left
			}
		}
		else = {
			add_prestige = prestige_weak_bonus
		}
	}
}
