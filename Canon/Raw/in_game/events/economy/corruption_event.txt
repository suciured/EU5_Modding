		scope:culture_location = {
			every_pop = {	add_pop_satisfaction = pop_satisfaction_severe_penalty	}
		}
	}
}

corruption_event.16 = {
	hide_portraits = yes
	type = country_event
	title = corruption_event.16.title
	desc = corruption_event.16.desc
	trigger = {
		"estate_power(estate_type:crown_estate)" < 0.1

		any_army = {
			any_sub_unit = {
				is_mercenary = yes
			}

			unit_location = {
				owner = root
				OR = {
					integration_level = integrated
					integration_level = core
				}
			}
		}
		at_war = no
	}

	illustration_tags = {
		10 = angry
		10 = exterior
	}

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
		hidden_effect = {
			random_army = {
				limit = {
					any_sub_unit = {
						is_mercenary = yes
					}	

					unit_location = {
						owner = root
						OR = {
							integration_level = integrated
							integration_level = core
						}
					}
				}
				unit_location = {
					save_scope_as = hired_mercs_victim_area
				}
				save_scope_as = hired_merc_band
			}
		}
	}

	option = {
		name = corruption_event.16.a
		scope:hired_merc_band = {
			every_sub_unit = {
				limit = {
					is_mercenary = yes
				}
				add_subunit_strength = {
					value = subunit_strength
					multiply = -0.10		
				}
				add_subunit_morale = {
					value = subunit_morale
					multiply = -0.50
				}
			}
		}
	}
	option = {
		name = corruption_event.16.b
		scope:hired_mercs_victim_area = {
			area = { 
				every_location_in_area = {
					change_prosperity = prosperity_very_weak_penalty
					change_development = development_weak_penalty
				} 
			}
		}
	}
}
