#Weather
qa_debug.1 = {
    type = country_event

    desc = qa_debug.1.desc

        always = no

			location = scope:end_location
		}
    }
    #Tornado
    option = {
        name = WEATHER_TORNADO
        start_weather_system = { 
			width = 800
			length = 100
			strength = 1.0
			speed = 30 
			type = tornado
			location = scope:start_location
			location = scope:end_location
		}
    }
    #Cylcone
    option = {
        name = WEATHER_CYCLONE
        start_weather_system = { 
			width = 800
			length = 100
			strength = 1.0
			speed = 30 
			type = cyclone
			location = scope:start_location

		}

#Heir Menu
qa_debug.1002 = {
	type = country_event
	title = qa_debug.1002.title
	desc = qa_debug.1002.desc
	immediate = {
		ruler ?= {
			if = {
				limit = { is_female = no }
				save_scope_as = ruler_father
			}
			else = {
				save_scope_as = ruler_mother
			}
		}
		consort ?= {
			if = {
				limit = { is_female = no }
				save_scope_as = consort_father
			}
			else = {
				save_scope_as = consort_mother
			}
		}
		if = {
			limit = { exists = scope:ruler_father }
			scope:ruler_father = { save_scope_as = father }
		}
		else_if = {
			limit = { exists = scope:consort_father }
			scope:consort_father = { save_scope_as = father }
		}
		if = {
			limit = { exists = scope:ruler_mother }
			scope:ruler_mother = { save_scope_as = mother }
		}
		else_if = {
			limit = { exists = scope:consort_mother }
			scope:consort_mother = { save_scope_as = mother }
		}
	}
	
	#A son of the [ROOT.GetCountry.GetGovernment.GetRuler.GetDynasty.GetName] dynasty
	option = {
		name = qa_debug.1002.a
		trigger = {
			ruler ?= { has_dynasty = yes }
			OR = {
				exists = scope:father
				exists = scope:mother
			}
		}
		if = {
			limit = {
				exists = scope:father
				exists = scope:mother
			}
			create_character = {
				save_scope_as = target_character
				dynasty = ruler.dynasty
				#father = scope:father
				mother = scope:mother
				female = no
			}
		}
		else_if = {
			limit = {
				exists = scope:father
			}
			create_character = {
				save_scope_as = target_character
				dynasty = ruler.dynasty
				father = scope:father
				female = no
			}
		}
		else = {
			create_character = {
				save_scope_as = target_character
				dynasty = ruler.dynasty
				mother = scope:mother
				female = no
			}
		}
		scope:target_character = { move_country = root }
	}
	#A daughter of the [ROOT.GetCountry.GetGovernment.GetRuler.GetDynasty.GetName] dynasty
	option = {
		name = qa_debug.1002.b
		trigger = {
			ruler ?= { has_dynasty = yes }
			OR = {
				exists = scope:father
				exists = scope:mother
			}
		}
		if = {
			limit = {
				exists = scope:father
				exists = scope:mother
			}
			create_character = {
				save_scope_as = target_character
				dynasty = ruler.dynasty
				father = scope:father
				mother = scope:mother
				female = yes
			}
		}
		else_if = {
			limit = {
				exists = scope:father

scripted_effect qa_create_character_with_dynasty = {
	if = {
		limit = { exists = scope:target_dynasty }
		create_character = {
			save_scope_as = target_character
			estate = $estate$
			female = $female$
			age = 20
			dynasty = scope:target_dynasty
		}
		hidden_effect = {
			if = {
				limit = {
					scope:target_dynasty = {
						any_character_in_dynasty = {
							is_adult = yes
							is_female = no
							NOT = { this = scope:target_character }
						}
					}
				}
				scope:target_dynasty = {
					random_character_in_dynasty = {
						limit = {
							is_adult = yes
							is_female = no
							NOT = { this = scope:target_character }
						}
						save_scope_as = new_dad
					}
				}
				scope:target_character = {
					change_father = scope:new_dad
				}
			}
			if = {
				limit = {
					NOT = { exists = scope:new_dad }
					scope:target_dynasty = {
						any_character_in_dynasty = {
							is_adult = yes
							is_female = yes
							NOT = { this = scope:target_character }
						}
					}
				}
				scope:target_dynasty = {
					random_character_in_dynasty = {
						limit = {
							is_adult = yes
							is_female = yes
							NOT = { this = scope:target_character }
						}
						save_scope_as = new_mom
					}
				}
				scope:target_character = {
					change_mother = scope:new_mom
				}
			}
		}
	}
	else = {
		create_character = {
			save_scope_as = target_character
			estate = $estate$
			female = $female$
			age = 20
		}
	}
}
