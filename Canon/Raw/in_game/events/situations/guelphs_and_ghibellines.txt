guelphs_and_ghibellines.10 = {
	type = country_event
	category = situation_event	
	title = guelphs_and_ghibellines.10.title
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = scope:winner_io
					exists = scope:loser_io
				}
				desc = guelphs_and_ghibellines.10.desc.specific
			}
			triggered_desc = {
				trigger = { always = yes }
				desc = guelphs_and_ghibellines.10.desc
			}
		}
	}
	major = yes
	major_trigger = {		
		OR = {
			international_organization:guelphs_io = { has_member = root }
			international_organization:ghibellines_io = { has_member = root }
		}
	}
	illustration_tags = {
		10 = armed
		10 = exterior
	}
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		if = {
			limit = { global_var:guelphs_and_ghibellines_ended = 1 }
			international_organization:guelphs_io = { save_scope_as = winner_io }
			international_organization:ghibellines_io = { save_scope_as = loser_io }
		}
		else_if = {
			limit = { global_var:guelphs_and_ghibellines_ended = -1 }
			international_organization:ghibellines_io = { save_scope_as = winner_io }
			international_organization:guelphs_io = { save_scope_as = loser_io }
		}
	}

	option = {
		name = guelphs_and_ghibellines.10.a
		destroy_international_organization = { target = international_organization:ghibellines_io }
		destroy_international_organization = { target = international_organization:guelphs_io }
	}
}

guelphs_and_ghibellines.100 = {
	type = country_event
	category = situation_event	
	title = guelphs_and_ghibellines.100.title
	desc = guelphs_and_ghibellines.100.desc
	illustration_tags = {
		10 = regular
		10 = interior
	}
	immediate = {
		save_scope_as = target_country
		ruler ?= { save_scope_as = target_character }
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
	}
	#We shall get ourselves involved in the matters of the Ghibellines
	option = {
		name = guelphs_and_ghibellines.100.a
		trigger = {
			NOT = { is_leader_of_international_organization = international_organization:ghibellines_io }
		}
		change_imperial_authority = { value = imperial_authority_weak_bonus }
		international_organization:ghibellines_io = {
			if = {
				limit = { scope:target_country = { NOT = { is_member_of_international_organization = international_organization:ghibellines_io } } }
				add_country_to_international_organization = scope:target_country
			}
			set_leader_country = scope:target_country
		}
		if = {
			limit = {
				scope:target_character = {
					has_character_modifier = guelphs_support_imperial_rivals_modifier
				}
			}
			scope:target_character ?= {
				remove_character_modifier = guelphs_support_imperial_rivals_modifier
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				NOT = { has_presence_in = region:italy_region }
			}
		}
	}
	option = {
		name = guelphs_and_ghibellines.100.c
		trigger = { is_leader_of_international_organization = international_organization:ghibellines_io }
		change_imperial_authority = { value = imperial_authority_weak_bonus }
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				NOT = { has_presence_in = region:italy_region }
			}
		}
	}
	#What the Italians are up to is of little interest to us
	option = {
		name = guelphs_and_ghibellines.100.b
		change_imperial_authority = { value = imperial_authority_mild_penalty }
		custom_tooltip = {
			text = guelphs_and_ghibellines.100.b.tt
			set_variable = gag_blocked_from_leading
		}
		if = {
			limit = { is_leader_of_international_organization = international_organization:ghibellines_io }
			international_organization_chooses_new_leader = international_organization:ghibellines_io
		}
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				has_presence_in = region:italy_region
			}
		}
	}
}

guelphs_and_ghibellines.1000 = {
	type = country_event
	category = situation_event	
	title = guelphs_and_ghibellines.1000.title
	desc = guelphs_and_ghibellines.1000.desc
	illustration_tags = {
		10 = regular
		10 = interior
	}
	immediate = { event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate } }
	#Accept
	option = {
		name = guelphs_and_ghibellines.1000.a
		trigger = {
			NOT = {
				scope:target_1 = {
					is_at_war_with = scope:actor
				}
			}
		}
		show_as_tooltip = {
			if = {
				limit = { scope:target.attacker_leader = scope:actor }
				scope:target_1 = { join_war_as_attacker = { war = scope:target } }
			}
			if = {
				limit = { scope:target.defender_leader = scope:actor }
				scope:target_1 = { join_war_as_defender = { war = scope:target } }
			}
		}
		hidden_effect = {
			scope:actor = {
				trigger_event_silently = guelphs_and_ghibellines.1001
			}
		}
	}
	#Refuse
	option = {
		name = guelphs_and_ghibellines.1000.b
		show_as_tooltip = {
			scope:actor = {
				add_opinion = {
					target = scope:target_1
					modifier = opinion_refuses_to_join_our_side
				}
			}
		}
		hidden_effect = {
			scope:actor = {
				trigger_event_silently = guelphs_and_ghibellines.1002
			}
		}
	}
}

guelphs_and_ghibellines.1100 = {
	type = country_event
	category = situation_event	
	title = guelphs_and_ghibellines.1100.title
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { exists = scope:target_country }
				desc = guelphs_and_ghibellines.1100.desc
			}
			triggered_desc = {
				trigger = { always = yes }
				desc = guelphs_and_ghibellines.1100.desc.fallback
			}
		}
	}
	illustration_tags = {
		10 = regular
		10 = interior
	}
	trigger = {
		international_organization:hre = {
			international_organization_has_leader = yes
		}
	}
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		save_scope_as = root_country
		international_organization:hre.leader_country ?= { save_scope_as = target_country }
		if = {
			limit = { exists = scope:target_country }
			set_variable = {
				name = manpower_to_transfer
				value = {
					value = scope:target_country.monthly_manpower
					multiply = 24
				}
			}
			set_variable = {
				name = gold_to_transfer
				value = {
					value = scope:target_country.yearly_gold
					multiply = 0.25
				}
			}
		}
	}
	after = {
		remove_variable = manpower_to_transfer
		remove_variable = gold_to_transfer
	}
	#Gold
	option = {
		name = guelphs_and_ghibellines.1100.a
		trigger = {
			exists = scope:target_country
			scope:target_country = {
				gold >= {
					value = yearly_gold
					multiply = 0.25
				}
			}
		}
		if = {
			limit = { exists = scope:target_country }
			custom_tooltip = {
				text = guelphs_and_ghibellines.1100.a.tt
				scope:target_country = {
					set_variable = {
						name = ghibellines_demand_gold
						value = yes
					}
					trigger_event_silently = guelphs_and_ghibellines.1101
				}
			}
		}
	}
	#Men
	option = {
		name = guelphs_and_ghibellines.1100.b
		trigger = {
			exists = scope:target_country
			var:manpower_to_transfer > 0
		}
		custom_tooltip = {
			text = guelphs_and_ghibellines.1100.b.tt
			scope:target_country = {
				set_variable = {
					name = ghibellines_demand_manpower
					value = yes
				}
				trigger_event_silently = guelphs_and_ghibellines.1101
			}
		}
	}
	#Alliance
	option = {
		name = guelphs_and_ghibellines.1100.c
		trigger = {
			exists = scope:target_country
			NOT = { is_allied_with = { target = scope:target_country } }
			NOT = { is_at_war_with = scope:target_country }
			NOT = { is_rival_of = scope:target_country }
			NOT = { is_enemy_of = scope:target_country }
		}
		custom_tooltip = {
			text = guelphs_and_ghibellines.1100.c.tt
			scope:target_country = {
				set_variable = {
					name = ghibellines_demand_alliance
					value = yes
				}
				trigger_event_silently = guelphs_and_ghibellines.1101
			}
