namespace = flavor_zzz

flavor_zzz.1 = {
	type = country_event
	title = flavor_zzz.1.title
	desc = flavor_zzz.1.desc

	fire_only_once = yes
	dynamic_historical_event = {
		tag = ZZZ
		from = 1400.1.1
		to = 1821.1.1
		monthly_chance = 1
	}

	trigger = {
		ruler_or_regent ?= {
            is_female = yes
        }
		location:zazzau.owner ?= root
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
		location:zazzau = {
			save_scope_as = target_location
		}

		ruler_or_regent ?= {
			save_scope_as = target_character
		}
	}

	option = {
		name = flavor_zzz.1.a
		
		scope:target_location = {
			construct_building = {
				building_type = building_type:zazzau_walls
			}
		}

        add_army_tradition = army_tradition_mild_bonus
	}

	option = {
		name = flavor_zzz.1.b
		
		add_prestige = prestige_mild_penalty
        change_societal_value = {
			type = offensive_vs_defensive
			value = societal_value_move_to_right
		}
	}
}

	}
}
