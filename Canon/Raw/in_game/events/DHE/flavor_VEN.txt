namespace = flavor_ven
flavor_ven.1 = {
	type = country_event
	title = flavor_ven.1.title
	desc = flavor_ven.1.desc
	fire_only_once = yes
	dynamic_historical_event = {
		tag = VEN
		from = 1400.1.1
		to = 1500.1.1
		monthly_chance = 10
	}
	trigger = {
		government_type = government_type:republic
		has_reform = government_reform:council_of_forty
	}

	illustration_tags = {
        10 = happy
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
	}

	option = {
		name = flavor_ven.1.a
		historical_option = yes
		unlock_government_reform_effect = { type = signoria_of_venice }
		if = {
			limit = { has_reform = government_reform:council_of_forty }
			remove_reform = government_reform:council_of_forty
		}
		else = { remove_current_major_government_reform = yes }
		add_reform = government_reform:signoria_of_venice
		add_stability = stability_weak_bonus
	}

	option = {
		name = flavor_ven.1.b
		unlock_parliament_issue_effect = { type = establish_the_signoria_of_venice }
		add_government_power = government_power_mild_bonus
	}
}
