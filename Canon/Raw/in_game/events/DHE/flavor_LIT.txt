flavor_lit.1 = {
	type = country_event
	title = flavor_lit.1.title
	desc = flavor_lit.1.desc

	fire_only_once = yes

	dynamic_historical_event = {
		tag = LIT
		from = 1550.1.1
		to = 1600.1.1
		monthly_chance = 5
	}

	trigger = {
		"religion_percentage_in_country(religion:lutheran)" > 0
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ordered_owned_location = {
			limit = {
				NOT = { has_building = building_type:lutheran_preachers }
			}

			order_by = population

			max = {
				value = 1
				add = {
					value = num_locations
					divide = 5
				}
			}
			every_pop = {
				limit = {
					NOT = { religion = religion:lutheran }
					religion.group = religion_group:christian
				}
				split_pop = {
					fraction = 0.10
					religion = religion:lutheran
				}
			}
		}
	}

	option = {
		name = flavor_lit.1.a
		historical_option = yes

		custom_tooltip = many_pops_will_become_lutheran_in_christian_locations_tt

		custom_tooltip = {
			text = lutheran_pops_gain_25_satisfaction
			every_owned_location = {
				limit = {
					any_pop = {
						owner = root
						religion = religion:lutheran
					}
				}
				every_pop = {
					limit = {
						owner = root
						religion = religion:lutheran
					}
					add_pop_satisfaction = pop_satisfaction_extreme_bonus
				}
			}
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_huge_move_to_right
		}
		add_government_power = government_power_extreme_penalty
		add_religious_influence_if_valid = { VALUE = religious_influence_extreme_penalty }
	}
	option = {
		name = flavor_lit.1.b
		custom_tooltip = many_pops_will_become_lutheran_in_christian_locations_tt

		custom_tooltip = {
			text = lutheran_pops_lose_33_satisfaction
			every_owned_location = {
				limit = {
					any_pop = {
						owner = root
						religion = religion:lutheran
					}
				}
				every_pop = {
					limit = {
						owner = root
						religion = religion:lutheran
					}
					add_pop_satisfaction = pop_satisfaction_extreme_penalty
				}
			}
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_huge_move_to_left

		scope:target_root_country.capital = { save_scope_as = target_location }

		scope:target_root_country.ruler = { save_scope_as = target_character }

	}

		historical_option = yes
		name = flavor_lit.3.a

			modifier = trust_treaty_of_ugra
			target = scope:target_root_country
		}

		create_relation = {
			first = this
			second = scope:target_root_country
			type = relation_type:alliance
		}

		custom_tooltip = {
			text = smolensk_owned_by_muscovy_tt
			area:smolensk_area = {
				every_location_in_area = {
					limit = {
						top_owner = scope:target_root_country
					}
					remove_core = this.owner
					change_location_owner = root
					add_core = root
				}
			}
		}

		scope:target_root_country = { trigger_event_silently = { id = flavor_mos.25 days = 0 } }

	}

	option = {
		name = flavor_lit.3.b

flavor_lit.5 = {
	hide_portraits = yes
	type = country_event

	title = flavor_lit.5.title
	desc = flavor_lit.5.desc
	historical_info = flavor_lit.5.historical_info
	fire_only_once = yes
	dynamic_historical_event = {
		tag = LIT
		from = 1370.1.1
		to = 1530.1.1
		monthly_chance = 5
	}

	trigger = {
		religion = religion:catholic
		ruler_or_regent = { has_available_marriage_slot = yes }
		is_subject = no
		at_war = no
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		ruler_or_regent = {
			save_scope_as = target_character
		}
		c:POL = {
			ruler_or_regent = {
				save_scope_as = target_character2
				if = {
					limit = {
						any_child = {
							is_alive = yes
							is_adult = yes
							has_available_marriage_slot = yes
							NOT = { is_same_gender = c:LIT.ruler }
							NOT = { is_close_relative = c:LIT.ruler }
							modifier:blocked_from_marriage = no
							NOT = {
								root.ruler_or_regent ?= {
									is_same_gender = PREV
								}
							}
						}
					}
					random_child = {
						limit = {
							is_alive = yes
							is_adult = yes
							has_available_marriage_slot = yes
							NOT = { is_same_gender = c:LIT.ruler }
							NOT = { is_close_relative = c:LIT.ruler }
							modifier:blocked_from_marriage = no
							NOT = {
								root.ruler_or_regent ?= {
									is_same_gender = PREV
								}
							}
						}
						save_scope_as = target_character3
					}
				}
				else = {
					prev = {
						if = {
							limit = {
								root.ruler_or_regent ?= {
									is_female = no
								}
							}
							create_character = {
								estate = estate_type:nobles_estate
								dynasty = ruler_or_regent.dynasty
								father = scope:target_character2
								female = yes
								age = 20
								save_scope_as = target_character3
							}
						}
						else = {
							create_character = {
								estate = estate_type:nobles_estate
								dynasty = ruler_or_regent.dynasty
								father = scope:target_character2
								female = no
								age = 20
								save_scope_as = target_character3
							}
						}
					}
				}
			}
		}
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
	}

	option = {
		name = flavor_lit.5.a

		historical_option = yes

		add_government_power = government_power_extreme_bonus

		hidden_effect = {
			set_variable = accepted_marriage_variable
		}
		custom_tooltip = lit_should_poland_accept_tt
	}

	option = {
		name = flavor_lit.5.b

		add_government_power = government_power_extreme_penalty
		c:POL = {
			add_casus_belli = {
				target = ROOT
				type = casus_belli:cb_claim_throne
				years = 10
			}
		}
	}

	after = {
		c:POL = { trigger_event_non_silently = flavor_lit.11 }
	}
