namespace = consort_events
consort_events.1 = {
	type = country_event
	title = consort_events.1.title
	desc = consort_events.1.desc
	trigger = {
		government_type = government_type:monarchy
		has_ruler_in_country = yes
		has_consort = no
		has_regent = no
		ruler ?= { 
			has_dynasty = yes
			is_female = no 
			is_adult = yes
			has_available_marriage_slot = yes
		}
		num_of_non_rural > 1
		estate_satisfaction:nobles_estate >= 0.5
		"estate_power(estate_type:nobles_estate)" > 0.25
		any_dynasty = {
			trigger_if = {
				limit = {
					root = {
						has_ruler_in_country = yes
						ruler = { has_dynasty = yes }
					}
				}
				root.ruler ?= { dynasty != prev }
			}
			any_character_in_dynasty = {
				is_female = yes
				is_adult = yes
				age_in_years < 30
				has_available_marriage_slot = yes
				is_close_relative = root.ruler
			}
		}
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ruler ?= { save_scope_as = target_ruler }
		random_dynasty = {
			limit = {
				root.ruler ?= { dynasty != prev }
				any_character_in_dynasty = {
					is_female = yes
					is_adult = yes
					age_in_years < 30
					has_available_marriage_slot = yes
					is_close_relative = root.ruler
				}
			}
			random_character_in_dynasty = {
				limit = {
					is_female = yes
					is_adult = yes
					age_in_years < 30
					has_available_marriage_slot = yes
					is_close_relative = root.ruler
				}
				save_scope_as = target_consort
			}
			dynasty_home = {
				save_scope_as = origin_of_queens_family
			}
		}
	}

	option = {
		name = consort_events.1.a
		custom_tooltip = consort_events.1.a.tt
		change_gold_effect = { scale = 2 }
		
		ruler ?= {
			marry_character = scope:target_consort
		}
	
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_bonus }
		
		hidden_effect = {
			random_list = {
				10 = { trigger_event_silently = { id = consort_events.2 days = { 720 2000 } } }
				10 = { trigger_event_silently = { id = consort_events.3 days = { 720 2000 } } }
				10 = { trigger_event_silently = { id = consort_events.4 days = { 720 2000 } } }
			}

		}
		
	}
	option = {
		name = consort_events.1.b
		add_prestige = prestige_mild_bonus
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_weak_penalty }
	}
}

consort_events.6 = {
	type = country_event
	title = consort_events.6.title
	desc = consort_events.6.desc

	trigger = {
		has_ruler_in_country = yes
		government_type = government_type:monarchy
		has_consort = no
		ruler ?= { is_adult = yes is_married = no}
	}
	
	illustration_tags = {
        10 = happy
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:nobles_estate }
		ruler ?= { save_scope_as = target_ruler }
		if = {
			limit = { 
				ruler ?= { is_female = no }
			}
			create_character = {
				culture = root.culture
				religion = root.religion
				age = 21
				female = yes
				create_in_limbo = yes
				save_scope_as = target_character
			}
		}
		else = {
			create_character = {
				culture = root.culture
				religion = root.religion
				age = 21
				female = no
				create_in_limbo = yes
				save_scope_as = target_character
			}
		}
	}

	option = {
		name = consort_events.6.a
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_penalty }
		hidden_effect = {
			scope:target_character = { move_country = root }
		}

		ruler ?= {
			marry_character = scope:target_character 
		}
		add_legitimacy = legitimacy_weak_penalty
		add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_mild_bonus }
	}
	option = {
		name = consort_events.6.b
		hidden_effect = { 
			kill_character_silently = scope:target_character
		}
		add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_mild_penalty }
	}
}

	type = country_event
	title = consort_events.61.title
	desc = consort_events.61.desc
	trigger = {
		government_type = government_type:monarchy
		has_ruler_in_country = yes
		has_consort = yes
		ruler = { owner = root }
		consort = {
			owner = root
			NOT = {
				religion = root.religion
			}
		}
		
	}
	
	illustration_tags = {
        10 = angry
        10 = interior
    }
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		consort = {
			save_scope_as = target_consort
		}
		ruler ?= {
			save_scope_as = target_ruler
		}
		consort.religion = {
			save_scope_as = origin_religion
		}
	
		ordered_owned_location = {
			limit = {
				integration_level = core
				NOT = { dominant_religion = scope:origin_religion }
			}
			order_by = population
			max = 1
			
			save_scope_as = province_under_religious_influence
		}
		
	}

	option = {
		name = consort_events.61.a
		add_legitimacy = legitimacy_mild_penalty
		scope:target_consort = { 
			if = {
				limit = { religion != root.religion }
				change_character_religion = root.religion
			}
		}
	}
	option = {
		name = consort_events.61.b
		trigger = {
			exists = scope:province_under_religious_influence
		}
		scope:province_under_religious_influence = {
			every_pop = {
				limit = {
					NOT = { religion = scope:origin_religion }
				}
				split_pop = {
					fraction = 0.25
					religion = scope:origin_religion
				}
			}
		}
	}
}
