artist_events.1 = { #Theological Works
	type = country_event
	title = artist_events.1.title
	desc = artist_events.1.desc
	illustration_tags = {
		10 = angry
		10 = interior
	}
	trigger = {
		any_artist = { artist_events_1_character_trigger = yes }
		country_has_estate = estate_type:clergy_estate
		country_has_estate = estate_type:peasants_estate
	}

	immediate = {
		random_artist = {
			limit = { artist_events_1_character_trigger = yes }
			save_scope_as = target_character
		}
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
	}

	option = {
		name = artist_events.1.a 

		change_societal_value = { type = traditionalist_vs_innovative value = societal_value_minor_move_to_right }
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_penalty }
		change_gold_effect = { scale = -1 }
		scope:target_character ?= {
			add_artist_skill = 0.03
		}
	}

	option = {
		name = artist_events.1.b 

		change_societal_value = { type = traditionalist_vs_innovative value = societal_value_minor_move_to_left }
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_weak_bonus }
		scope:target_character ?= {
			add_artist_skill = -0.05
		}
	}

	option = {
		name = artist_events.1.c 

		trigger = {
			OR = {
				societal_value:traditionalist_vs_innovative < 50
				ruler ?= { 
					OR = {
						has_trait = cruel 
						has_trait = zealot 
					} 
					NOT = { has_trait = free_thinker }
				}
			}
		}

		change_societal_value = { type = traditionalist_vs_innovative value = societal_value_minor_move_to_left }
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
		kill_character = {
			target = scope:target_character
			reason = beheading
			killer = ruler
		}
	}

	option = {
		name = artist_events.1.d 

		trigger = {
			OR = {
				societal_value:traditionalist_vs_innovative > 50
				ruler ?= { 
					OR = {
						has_trait = scholar 
						has_trait = free_thinker
						has_trait = sinner 
					} 
					NOT = { has_trait = zealot }
				}
			}
		}
		change_gold_effect = { scale = -2 }
		change_societal_value = { type = traditionalist_vs_innovative value = societal_value_minor_move_to_left }
		every_artist = { add_artist_skill = 0.05 }
	}
}

artist_events.6 = { #The Wall Art
	type = country_event
	title = artist_events.6.title
	desc = artist_events.6.desc
	illustration_tags = {
		10 = happy
		10 = interior
	}
	trigger = {
		any_artist = {
			artist_type = painter
		}
		any_owned_location = {
			is_capital = no
			population >= 10 
		}
	}

	immediate = {
		random_owned_location = {
			limit = {
				is_capital = no
				population >= 10 
			}
			save_scope_as = target_location
		}
		random_artist = {
			limit = {
				artist_type = painter 
			}
			save_scope_as = target_character
		}	
		event_illustration_government_estate_effect = yes
	}

	option = {
		name = artist_event.6.a 

		change_gold_effect = { scale = -1 }
		scope:target_location = {
			if = {
				limit = { development <= 20 }
				change_development = development_weak_bonus
			}
			else = {
				every_pop = {
					limit = {
						owner = root
					}
					add_pop_satisfaction = pop_satisfaction_weak_bonus
				}
			}
			change_prosperity = prosperity_very_weak_bonus
		}
		scope:target_character = {
			if = {
				limit = { artist_skill >= 0.98 }
				add_trait = trait:prominent
			}
			else = { add_artist_skill = 0.03 }
		}
	}

	option = {
		name = artist.events.6.b 

		add_prestige = prestige_weak_penalty
	}
}

				}

		}

			add_artist_skill = 0.2
		}

		}
