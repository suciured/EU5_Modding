personality_events.10 = {
	type = country_event
	title = personality_events.10.title
	desc = personality_events.10.desc
	illustration_tags = {
		10 = angry
		10 = interior
	}
	trigger = {
		country_has_estate = estate_type:nobles_estate
		country_has_estate = estate_type:clergy_estate
		has_ruler = yes
		ruler = { has_trait = just }
		NOT = { has_country_modifier = just_settlement_modifier }
		any_province = {
			province_average_control >= 0
			province_average_integration >= 0.5
			NOT = { has_province_modifier = increasing_local_authority }
		}
	}

	immediate = {
		random_province = {
			limit = {
				province_average_control >= 0
				province_average_integration >= 0.5
				NOT = { has_province_modifier = increasing_local_authority }
			}
			save_scope_as = target_province
		}

		ruler = { save_scope_as = target_ruler }
		event_illustration_poptype_effect = { foreground = pop_type:nobles   background = pop_type:nobles }
	}

	option = {
		name = personality_events.10.a
		add_estate_satisfaction = {
			type = estate_type:nobles_estate
			value = estate_satisfaction_mild_bonus
		}
		add_country_modifier = { modifier = just_settlement_modifier years = 5 mode = add_and_extend }
	}
	option = {
		name = personality_events.10.b
		trigger = {
			NOT = { ruler ?= { has_trait = greedy } }
		}

		scope:target_province ?= {
			add_province_modifier = { modifier = increasing_local_authority years = 5 mode = add_and_extend }
		}
	}
}

personality_events.15 = {
	type = country_event
	title = personality_events.15.title
	desc = personality_events.15.desc
	illustration_tags = {
		10 = angry
		10 = interior
	}
	trigger = {
		country_has_estate = estate_type:clergy_estate
		has_ruler = yes
		ruler = {
			has_trait = free_thinker
			NOT = { has_trait = babbling_buffoon }
			NOT = { has_trait = zealot }
			NOT = { has_character_modifier = inspiring_spiritual_guide }
		}

		stability < 60
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		ruler = { save_scope_as = target_ruler }
		create_character = {
			age = { 20 70 }
			female = no
			random_estate_type_trigger = {
				root = { estate_type_allowed_in_cabinet = {	estate_type = prev } }
			}
			create_in_limbo = yes

			save_scope_as = heresiarch
		}
	}

	option = {
		name = personality_events.15.a
		add_estate_satisfaction = { type = estate_type:clergy_estate
			value = estate_satisfaction_mild_penalty
		}
		ruler = { add_character_modifier = { modifier = inspiring_spiritual_guide months = -1 mode = add_and_extend } }

		scope:heresiarch = { move_country = root }
	}
	option = {
		name = personality_events.15.b
		add_estate_satisfaction = { type = estate_type:clergy_estate
			value = estate_satisfaction_mild_bonus
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_tiny_move_to_left
		}

		kill_character = {
			target = scope:heresiarch
			reason = execution
			killer = ruler
		}
	}
	option = {
		name = personality_events.15.c
		add_prestige = prestige_weak_penalty
		add_estate_satisfaction = { type = estate_type:clergy_estate
			value = estate_satisfaction_weak_penalty
		}

		hidden_effect = { kill_character_silently = scope:heresiarch }
	}
}
