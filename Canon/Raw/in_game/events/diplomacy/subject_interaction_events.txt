namespace = subject_interaction_events
subject_interaction_events.1 = {
	hide_portraits = yes
	type = country_event
	title = subject_interaction_events.1.title
	desc = subject_interaction_events.1.desc
	trigger = {
		is_subject = yes
		
		government_type = government_type:monarchy
		
		exists = previous_ruler 
		exists = ruler
		exists = ruler.dynasty
		exists = previous_ruler.dynasty
		NOT = { previous_ruler.dynasty = ruler.dynasty }
		
		monthly_manpower > 0
		
		overlord = {
			ruler ?= { dynasty ?= root.ruler.dynasty }
			root = {
				NOR = { 
					has_opinion = { target = prev modifier = opinion_sent_help }
					has_opinion = { target = prev modifier = opinion_refused_send_help }
				}
			}
		}
	}

	immediate = {
		previous_ruler = { save_scope_as = target_previous_ruler }
		ruler = { save_scope_as = target_ruler }
	}

	option = {
		name = subject_interaction_events.1.a
		random_owned_location = {
			limit = {
				is_capital = yes
			}
			every_pop = {	add_pop_satisfaction = pop_satisfaction_severe_penalty	}
		}
		
		save_scope_as = subject_in_need
		custom_tooltip = subject_interaction_events.ask_overlord_for_help
		overlord = {
			trigger_event_silently = { id = subject_interaction_events.2 }
		}
	}
	option = {
		name = subject_interaction_events.1.b
		add_prestige = prestige_mild_bonus
		random_owned_location = {
			limit = {
				is_capital = yes
			}
			every_pop = {	add_pop_satisfaction = pop_satisfaction_mild_penalty	}
		}
	}
}

subject_interaction_events.2 = {

	desc = subject_interaction_events.2.desc
	option = {

			set_variable = { name = si_pretender_help_flag value = 1 }

		}
	}
}

scripted_trigger is_bad_relative_ruler = {
	has_regent = no
		has_ruler = yes
		government_type = government_type:monarchy
		ruler = { 
			dynasty ?= root.ruler.dynasty 
			this != root.ruler
			adm < 50
			dip < 50
			mil < 50
		}
		any_character = {
			is_alive = yes
			is_adult = yes
			dynasty ?= root.ruler.dynasty
			this != root.ruler
			OR = {
				adm > 50
				dip > 50
				mil > 50
			}
		}
}

subject_interaction_events.4 = {
	type = country_event
	title = subject_interaction_events.4.title
	desc = subject_interaction_events.4.desc
	trigger = {
		
		has_regent = no
		has_ruler = yes
		ruler ?= { has_dynasty = yes }
		any_subject = {
			is_bad_relative_ruler = yes
		}
	}

	immediate = {
		random_subject = {
			limit = {
				is_bad_relative_ruler = yes
			}
			save_scope_as = subject_target
			
			random_character = {
				limit = {
					is_alive = yes
					is_adult = yes
					dynasty ?= root.ruler.dynasty
					this != root.ruler
					OR = {
						adm > 50
						dip > 50
						mil > 50
					}
				}
				save_scope_as = better_ruler
			}
		}
	}


	option = {
		name = subject_interaction_events.4.a
		trigger = {
			exists = scope:subject_target
		}
		scope:subject_target = {
			set_new_ruler = scope:better_ruler
		}
	}
	option = {
		name = subject_interaction_events.4.b
		add_legitimacy = legitimacy_weak_penalty
	}
}
