namespace = colonial_nation


colonial_nation.4 = {
	type = country_event
	title = colonial_nation.4.title
	desc = colonial_nation.4.desc
	trigger = {
		is_subject_type = colonial_nation
		liberty_desire > 50
		
		OR = {
			current_age = age_5_absolutism
			current_age = age_6_revolutions
		}
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
		save_scope_as = from
		create_character = {
			trait_category = general
			save_scope_as = new_general_scope
			create_in_limbo = yes
		}
	}

	option = {
		name = colonial_nation.4.a
		scope:new_general_scope = { move_country = root }

		overlord = {
			trigger_event_silently = { id = colonial_nation.5 }
		}
	}
	option = {
		name = colonial_nation.4.b
		add_liberty_desire = liberty_desire_mild_plus
		overlord = {
			trigger_event_silently = { id = colonial_nation.6 }
		}
		kill_character = {
			target = scope:new_general_scope
			reason = execution
			killer = ruler
		}
	}
}

colonial_nation.14 = {
	hide_portraits = yes
	type = country_event
	title = colonial_nation.14.title
	desc = colonial_nation.14.desc
	trigger = {
		any_subject = {
			is_subject_type = colonial_nation
		}
		war_exhaustion > 4
	}
	
	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
		save_scope_as = from
	}

	option = {
		name = colonial_nation.14.a
		add_war_exhaustion = war_exhaustion_mild_bonus
		ordered_subject = {
			limit = {
				is_subject_type = colonial_nation
			}
			order_by = total_population
			max = {
				value = 1
				add = {
					value = num_subjects
					divide = 3
				}
			}
			
			show_as_tooltip = {
				add_liberty_desire = liberty_desire_mild_plus
				add_war_exhaustion = war_exhaustion_weak_penalty
			}
			trigger_event_silently = { id = colonial_nation.15 }
		
		}
	}
	option = {
		name = colonial_nation.14.b
		add_prestige = prestige_mild_penalty
	}
}
