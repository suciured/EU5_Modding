conquest_of_paradise.2 = {
	hide_portraits = yes
	type = country_event
	title = conquest_of_paradise.2.title
	desc = conquest_of_paradise.2.desc
	trigger = {
		is_subject_type = colonial_nation
		any_neighbor_country = {
			is_subject_type = colonial_nation
			overlord = root.overlord
		}
		manpower > 0 
		
		overlord = {
			has_ruler = yes
		}
	}
	image = "gfx/interface/illustrations/institutions/new_world.dds"
	immediate = {
		random_neighbor_country = {
			limit = {
				is_subject_type = colonial_nation
				overlord = root.overlord
			}
			save_scope_as = colonial_conflict_target
			ordered_province = {
				order_by = population
				max = 1
				save_scope_as = source_province
			}
		}
	}


	option = {
		name = conquest_of_paradise.2.a
		custom_tooltip = conquest_of_paradise.2.a.tt
		save_scope_as = colonial_conflict_source
		scope:colonial_conflict_target = {
			trigger_event_silently = { id = conquest_of_paradise.3 }
		}
		add_manpower = { value = root.monthly_manpower multiply = -3 }
	}
	option = {
		name = conquest_of_paradise.2.b
		add_prestige = prestige_mild_penalty

		add_migration = {
			owner = this
			from = scope:source_province.province_definition
			to = capital.province_definition
			type = pop_type:peasants
			amount = 0.025
			months = 36
		}

	}
}

conquest_of_paradise.15 = {
	type = country_event
	title = conquest_of_paradise.15.title
	desc = conquest_of_paradise.15.desc
	trigger = {
		is_subject_type = colonial_nation
		OR = {
			produced_in_country:sugar > produced_in_country:wheat
			produced_in_country:cotton > produced_in_country:wheat
			produced_in_country:tobacco > produced_in_country:wheat
		}
		any_owned_location = {
			OR = {
				raw_material = goods:sugar
				raw_material = goods:cotton
				raw_material = goods:tobacco
			}
			rgo_workers > 2
			
		}
	}

	image = "gfx/interface/illustrations/institutions/new_world.dds"
	immediate = {
		random_owned_location = {
			limit = {
				OR = {
					raw_material = goods:sugar
					raw_material = goods:cotton
					raw_material = goods:tobacco
				}
				rgo_workers > 2
			}
			save_scope_as = cash_crop_location
		}
	}

	option = {
		name = conquest_of_paradise.15.a
		scope:cash_crop_location = { 
			change_max_raw_material_workers = -2 
		}
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_penalty }
		add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_extreme_bonus }
	}
	option = {
		name = conquest_of_paradise.15.b
		add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_extreme_penalty }
	}
}

	type = country_event
	title = conquest_of_paradise.16.title
	desc = conquest_of_paradise.16.desc
	
	image = "gfx/interface/icons/trade_goods/illustrations/icon_goods_tobacco.dds"
	
	trigger = {
		is_subject_type = colonial_nation
		capital.market = { "market_price(goods:tobacco)" < "default_price(goods:tobacco)" }
