is_proposal = {
	custom_description = {
		text = is_proposal_text
		subject = $resolution$
		object = $international_organization$
		exists = scope:actor
		exists = $international_organization$
		OR = {
			NOT = { exists = scope:proposer }
			scope:proposer = scope:actor
		}
		#$international_organization$ = {
		#	NOT = { resolution_is_active = $resolution$ }
		#}
	}
}

is_vote = {
	custom_description = {
		text = is_vote_text
		subject = $resolution$
		object = $international_organization$
		exists = scope:actor
		exists = scope:proposer
		scope:proposer != scope:actor
		exists = $international_organization$
		#$international_organization$ = {
		#	resolution_is_active = $resolution$
		#}
	}
}

is_finalising = {
	custom_description = {
		text = is_finalising_text
		subject = $resolution$
		object = $international_organization$
		not = { exists = scope:actor }
		exists = scope:proposer
		exists = $international_organization$
	}
}
