levy_fishing_boat = {
	size = levy_ship_size
	
	#these guys own the boats. 1 burgher per boat
	allow = {
		pop_type = pop_type:burghers
	}

	#crew members. Number per boat is set by the sub unit definition, in this case n_fishing_boat
	allow_as_crew = {
		pop_type = pop_type:peasants
	}

	unit = n_fishing_boat
}

levy_birlinn = {
	size = levy_ship_size_special
	
	#these guys own the boats. 1 noble per boat
	allow = {
		pop_type = pop_type:nobles
		culture.language = language:gaelic_language
	}

	#crew members. Number per boat is set by the sub unit definition
	allow_as_crew = {
		OR = {
			pop_type = pop_type:peasants
			pop_type = pop_type:tribesmen
			pop_type = pop_type:laborers
			pop_type = pop_type:slaves
		}
	}

	unit = n_birlinn
}
