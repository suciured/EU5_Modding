random_event.18 = {
	type = country_event

	title = random_event.18.title
	desc = random_event.18.desc
	illustration_tags = {
		
		10 = interior
	}

	trigger = {
		has_ruler = yes
		has_regent = no
		exists = capital

		any_neighbor_country = {
			at_war = no
			is_subject = no
			exists = capital
			capital.continent ?= root.capital.continent
			NOT = { has_casus_belli_on = root }
			NOT = { has_opinion = { target = root modifier = opinion_good_relations } }
			has_ruler = yes
			NOT = { ruler = root.ruler }
		}
	}
	
	immediate = {
		random_neighbor_country = {
			limit = {
				at_war = no
				is_subject = no
				exists = capital
				capital.continent ?= root.capital.continent
				NOT = { has_casus_belli_on = root }
				NOT = { has_opinion = { target = root modifier = opinion_good_relations } }
				has_ruler = yes
				NOT = { ruler = root.ruler }
			}
			save_scope_as = insulted_party
			ruler = {save_scope_as = angry_ruler}
		}
		ruler = {save_scope_as = target_ruler}
		
	}
	
	option = {
		name = random_event.18.a
		scope:insulted_party = {
			add_casus_belli = { target = root type = casus_belli:cb_insulted_us }
		}
	}

	option = {
		name = random_event.18.b
		add_legitimacy = legitimacy_mild_penalty
		scope:insulted_party = {
			add_opinion = { target = root modifier = opinion_good_relations }
		}
	}
}

	title = random_event.22.title

		10 = happy
		10 = exterior
	}

	trigger = {
		country_has_estate = estate_type:clergy_estate
		country_has_estate = estate_type:nobles_estate
		estate_tax_base:clergy_estate > 1.0
		religion = religion:catholic
		has_ruler = yes
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ruler = { save_scope_as = target_ruler }
	}

	option = {
		name = random_event.22.a
		add_gold = {
			value = root.estate_tax_base:clergy_estate
			multiply = 6
		}

		add_estate_satisfaction = { type = estate_type:nobles_estate
			value = estate_satisfaction_mild_bonus
		}
		add_gold_to_estate = {
			estate_type = estate_type:nobles_estate
			value = {
				value = root.estate_tax_base:clergy_estate
				multiply = 6
			}
		}

		add_estate_satisfaction = { 
			type = estate_type:clergy_estate
			value = estate_satisfaction_extreme_penalty
		}
		add_gold_to_estate = {
			estate_type = estate_type:clergy_estate
			value = {
				value = root.estate_tax_base:clergy_estate
				multiply = -12
			}
		}
	}

	option = {
		name = random_event.22.b
		add_stability = stability_mild_bonus
		add_estate_satisfaction = {
			type = estate_type:nobles_estate
			value = estate_satisfaction_weak_penalty
		}
		add_estate_satisfaction = {
			type = estate_type:clergy_estate
			value = estate_satisfaction_mild_bonus
		}
	}

random_event.39 = {
	type = country_event

	title = random_event.39.title
	desc = random_event.39.desc
	illustration_tags = {
		10 = armed
		10 = exterior
	}

	trigger = {
		any_neighbor_country = {
			NOT = { is_allied_with = { target = root } }
			NOT = { in_union_with = root }
			is_subject = no
			NOT = { has_opinion = { target = root modifier = opinion_supports_dissidents } }
			has_ruler = yes
			in_civil_war = no
		}
		country_has_estate = estate_type:nobles_estate
	}

	immediate = {
		random_neighbor_country = {
			limit = {
				NOT = { is_allied_with = { target = root } }
				NOT = { in_union_with = root }
				is_subject = no
				NOT = { has_opinion = { target = root modifier = opinion_supports_dissidents } }
				has_ruler = yes
				in_civil_war = no
			}
			save_scope_as = dissidents
		}
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
	}

	option = {
		name = random_event.39.a
		add_estate_satisfaction = { type = estate_type:nobles_estate
			value = estate_satisfaction_weak_bonus
		}
		change_gold_effect = { scale = -2 }

		scope:dissidents = { add_opinion = { target = root modifier = opinion_supports_dissidents } }
	}

	option = {
		name = random_event.39.b
		add_estate_satisfaction = { type = estate_type:nobles_estate
			value = estate_satisfaction_mild_penalty
		}
		add_prestige = prestige_mild_penalty
	}
}

	}
	
	trigger = {
		country_has_estate = estate_type:peasants_estate
		num_of_non_rural >= 1
		num_location_rank:rural_settlement >= 1
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		ordered_owned_rural_location = {
			order_by = development
			max = {
				value = 3
				add = {
					value = num_provinces
					multiply = 0.25
					ceiling = yes
				}
			}
			check_range_bounds = no
			add_to_list = list_of_rural
		}
		random_in_list = {
			list = list_of_rural
			save_scope_as = starting_location
		}
		ordered_owned_non_rural_location = {
			order_by = {
				#They are more likely to move to the closest urban location
				#And not across the continent or beyond
				value = {
					add = "distance_to_squared(scope:starting_location)"
					multiply = -1
				}
				#More likely to move to non-overpopulated towns and cities
				#However we don't want them to populate ghost-towns
				if = {
					limit = { location_population_percentage <= 0 }
					divide = 0.75
				}
				else = { divide = location_population_percentage }

				if = {
					limit = { has_variable = recently_received_serfs_variable }
					multiply = 0.1
				}
			}
			save_scope_as = target_location
		}
	}

	option = {
		name = random_event.68.a

		custom_tooltip = {
			text = random_event.68.a.tt

			scope:starting_location = {
				every_pop = {
					limit = { owner = root }
					add_pop_satisfaction = pop_satisfaction_severe_penalty
				}
			}
		}

		scope:starting_location = {
			change_development = development_mild_penalty
		}
		scope:target_location = {
			change_development = development_mild_bonus

			hidden_effect = {
				set_variable = {
					name = recently_received_serfs_variable
					years = 10
				}
			}
		}

		if = {
			limit = {
				scope:starting_location.province = {
					this != scope:target_location.province
					population > {
						value = 1
						add = {
							value = scope:target_location.province.population
							multiply = 0.25
						}
					}
				}
			}
			add_migration = {
				owner = root
				from = scope:starting_location.province_definition
				to = scope:target_location.province_definition
				amount = {
					value = {
						add = scope:target_location.province.population
						multiply = 0.01
					}
				}
				months = 5
			}
		}
		
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_minor_move_to_left
		}
	}
	option = {
		name = random_event.68.b

		add_estate_satisfaction = { 
			type = estate_type:peasants_estate
			value = estate_satisfaction_mild_penalty
		}
		add_estate_satisfaction = { 
			type = estate_type:nobles_estate
			value = estate_satisfaction_extreme_bonus
		}

		if = {
			limit = { has_estate_privilege = estate_privilege:peasants_free_peasantry }

			custom_tooltip = following_happens_because_of_peasants_free_peasantry_tt
			change_societal_value = {
				type = serfdom_vs_free_subjects
				value = societal_value_minor_move_to_left
			}
			add_stability = stability_weak_penalty
		}
		else = {
			change_societal_value = {
				type = centralization_vs_decentralization
				value = societal_value_minor_move_to_right
			}
		}
	}
}

scripted_trigger random_event_75_character_trigger = {
	is_adult = yes
	is_ruler = no
	is_close_relative = root.ruler
	total_abilities < 100
	dip < 50
	NOT = {
		exists = rebel
	}
}
random_event.75 = {
	type = country_event

	title = random_event.75.title
	desc = random_event.75.desc
	illustration_tags = {
		10 = angry
		10 = interior
	}
	
	trigger = {
		government_type = government_type:monarchy
		has_ruler = yes
		
		any_character = {
			random_event_75_character_trigger = yes
		}

		NOT = { has_country_modifier = compromised_counter_espionage }
	}

	immediate = {
		random_character = {
			limit = {
				random_event_75_character_trigger = yes
			}
			save_scope_as = naive
		}
		ruler = { save_scope_as = target_ruler }
		event_illustration_government_estate_effect = yes
	}

	option = {
		name = random_event.75.a
		kill_character_silently = scope:naive
		add_prestige = prestige_weak_penalty
	}
	option = {
		name = random_event.75.b
		add_country_modifier = {
			modifier = compromised_counter_espionage
			years = 10
			mode = add_and_extend
		}
	}
}

random_event.100 = {
	type = country_event

	title = random_event.100.title
	desc = random_event.100.desc
	illustration_tags = {
		10 = regular
		10 = interior
	}

	trigger = {
		any_market_with_merchants = {
			count > 3
		}
		capital ?= { continent = continent:europe }
		discovered_any_in_americas_trigger = no
		discovered_route_to_india = yes
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }
		
		create_character = {
			trait_category = explorer
			random_estate_type_trigger = {
				root = { estate_type_allowed_in_cabinet = {	estate_type = prev } }
			}
			create_in_limbo = yes
			
			save_scope_as = india_explorer
		}
	}

	option = {
		name = random_event.100.a

		scope:india_explorer = { move_country = root }
		add_prestige = prestige_mild_bonus
	}

	option = {
		name = random_event.100.b
		add_gold = {
			value = root.monthly_trade_income
			multiply = 6
		}
		add_stability = stability_mild_penalty
		hidden_effect = { kill_character_silently = scope:india_explorer }
	}
}

random_event.112 = { #Poor Copper
	type = country_event

	title = random_event.112.title
	desc = random_event.112.desc
	illustration_tags = {
		10 = angry
		10 = interior
	}

	trigger = {
		country_has_estate = estate_type:burghers_estate
		country_has_estate = estate_type:peasants_estate
		any_owned_location ?= {
			raw_material = goods:copper
		}
		produced_in_country:copper > 0
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }

		ordered_owned_location = {
			limit = { raw_material = goods:copper }
			order_by = development
			max = 1
			check_range_bounds = no
			save_scope_as = target_location
		}
	}

	option = { #Side with Merchant
		name = random_event.112.a
		add_estate_satisfaction = { 
			type = estate_type:burghers_estate
			value = estate_satisfaction_mild_bonus
		}

		scope:target_location = {
			ordered_pop = {
				limit = { 
					owner = root
					pop_type = pop_type:peasants 
				}
				order_by = pop_size
				max = 1
				check_range_bounds = no
				add_pop_satisfaction = pop_satisfaction_mild_penalty
			}
		}
	}

	option = { #Side with Peasant
		name = random_event.112.b

		add_estate_satisfaction = { 
			type = estate_type:peasants_estate
			value = estate_satisfaction_mild_bonus
		}

		change_societal_value = { 
			type = capital_economy_vs_traditional_economy
			value = societal_value_minor_move_to_right
		}
	}

	option = { #Let them resolve it among themselves
		name = random_event.112.c

		add_estate_satisfaction = { 
			type = estate_type:burghers_estate
			value = estate_satisfaction_weak_penalty
		}

		add_estate_satisfaction = { 
			type = estate_type:peasants_estate
			value = estate_satisfaction_weak_penalty
		}

		add_government_power = government_power_weak_bonus
	}
}
