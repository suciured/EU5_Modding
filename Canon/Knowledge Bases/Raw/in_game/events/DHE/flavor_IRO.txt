	option = { # Times are changing.
		name = flavor_iro.1.a

		custom_tooltip = may_unify_haudenosaunee_confederation
		set_variable = enabled_iroquois_unique_interactions

			if = {
				limit = { 
					country_exists = c:ONO

			}
			if = {
				limit = { 
					country_exists = c:KKA
					this != c:KKA
				}
				c:KKA = { trigger_event_silently = { id = flavor_iro.100 days = 1 } }
			}
			if = {
				limit = { 
					country_exists = c:ONY

				}
				c:ONY = { trigger_event_silently = { id = flavor_iro.100 days = 1 } }

flavor_iro.30 = { # Shikellamy and Pennsylvania
	type = country_event
	title = flavor_iro.30.title
	desc = flavor_iro.30.desc

	fire_only_once = yes
	dynamic_historical_event = {
		tag = IRO
		from = 1700.1.1
		to = 1750.1.1
		monthly_chance = 10
	}

	trigger = {
		has_ruler = yes
		area:pennsylvania_area = {
			any_location_in_area = {
				OR = {
					owner ?= { is_subject_of = root }
					owner ?= root
				}	
				count > 5			
			}
		}		
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ruler ?= {
            save_scope_as = our_ruler
        }	
		create_character = {
			first_name = shikellamy
			adm = 60
			dip = 50
			mil = 61
			culture = culture:onondowaga_culture

			birth_location = location:canadaigua # He was Huron or Onodowaga

			save_scope_as = target_character
			script = shikellamy_overseer
			estate = estate_type:nobles_estate
			create_in_limbo = yes
		}	
	}	
	
	illustration_tags = {
		10 = regular
		10 = interior
	}	

	option = { # Appoint him as Chief over <region_name>
		name = flavor_iro.30.a	
		historical_option = yes		
		set_variable = promoted_shikellamy
		scope:target_character = {
			move_country = ROOT						
		}		
		area:pennsylvania_area = {
			every_location_in_area = {
				limit = {
					owner ?= root

					modifier = iro_shikellamy_pennsylvania

			}

	}

	}
}

flavor_iro.31 = { # Shikellamy and the Walking Purchase
	type = country_event
	title = flavor_iro.31.title
	desc = flavor_iro.31.desc

	fire_only_once = yes
	dynamic_historical_event = {
		tag = IRO
		from = 1700.1.1
		to = 1776.1.1
		monthly_chance = 10
	}

	trigger = {
		has_ruler = yes
		exists = character:shikellamy_overseer
		has_variable = promoted_shikellamy
		character:shikellamy_overseer ?= {
			is_alive = yes
			owner = ROOT
		}
		area:pennsylvania_area = {
			any_location_in_area = {
				OR = {
					owner ?= { is_subject_of = root }
					owner ?= root
				}
				any_neighbor_location = {
					is_discovered_by = root
					owner ?= {
						has_colonial_charters = yes
						religion.group = religion_group:christian
					}
				}				
			}
		}
		NOT = {
			province_definition:delaware_province = {
				any_location_in_province_definition = {
					owner ?= ROOT
				}
			}		
		}	
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ruler ?= {
            save_scope_as = our_ruler
        }	
		character:shikellamy_overseer ?= {
			save_scope_as = target_character
		}
		random_neighbor_country = {
			limit = {
				has_colonial_charters = yes
				religion.group = religion_group:christian
			}
			save_scope_as = colonial_nation_target
		}					
	}	
	
	illustration_tags = {
		10 = regular
		10 = interior
	}	

	option = { # Agree to the Purchase with <colonial_name>
		name = flavor_iro.31.a	
		historical_option = yes	
		add_opinion = { target = scope:colonial_nation_target modifier = iro_walking_sale }
		reverse_add_opinion = { target = scope:colonial_nation_target modifier = iro_walking_sale }	
		add_gold = {
			value = scope:colonial_nation_target.monthly_income_trade_and_tax
			multiply = 4
		}	
		province_definition:delaware_province = {
			every_location_in_province_definition = {
				limit = {
					NOT = { owner ?= scope:colonial_nation_target }
				}
				add_core = scope:colonial_nation_target
			}
		}			
	}
	
	option = { # We cannot agree to this proposal!
		name = flavor_iro.31.b	
		add_prestige = prestige_mild_penalty
		add_stability = stability_mild_bonus
		add_opinion = { target = scope:colonial_nation_target modifier = refused_iro_walking_sale }	
		reverse_add_opinion = { target = scope:colonial_nation_target modifier = refused_iro_walking_sale }			
	}
}
