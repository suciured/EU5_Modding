scripted_effect lots_of_sikhs = {
	capital.area = {
		every_location_in_area = {
			limit = { owner ?= root }
			every_pop = {
				limit = {
					OR = {
						religion.group = religion_group:muslim
						religion = religion:hindu
					}
				}
				random_list = {
					10 = {}
					40 = {
						split_pop = {
							fraction = 0.05
							religion = religion:sikhism
						}
					}
					20 = {
						split_pop = {
							fraction = 0.10
							religion = religion:sikhism
						}
					}
					15 = {
						split_pop = {
							fraction = 0.20
							religion = religion:sikhism
						}
					}
					10 = {
						split_pop = {
							fraction = 0.30
							religion = religion:sikhism
						}
					}
					4 = {
						split_pop = {
							fraction = 0.40
							religion = religion:sikhism
						}
					}
					1 = {
						split_pop = {
							fraction = 0.50
							religion = religion:sikhism
						}
					}
				}
			}
		}
	}
}

sikhism.1 = {
	type = country_event
	title = sikhism.1.title
	desc = sikhism.1.desc
	fire_only_once = yes

	trigger = {
		current_year > 1509
		any_owned_location = {
			OR = {
				AND = {
					dominant_religion ?= religion:hindu
					any_neighbor_location = {
						dominant_religion ?= {
							group = religion_group:muslim
						}
					}
				}
				AND = {
					dominant_religion ?= {
						group = religion_group:muslim
					}
					any_neighbor_location = {
						dominant_religion ?= religion:hindu
					}
				}
			}
		}
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		religion:sikhism = { enable_religion = yes }
		ordered_owned_location = {
			order_by = population
			limit = {
				dominant_religion != religion:sikhism
				OR = {
					AND = {
						dominant_religion ?= religion:hindu
						any_neighbor_location = {
							dominant_religion ?= {
								group = religion_group:muslim
							}
						}
					}
					AND = {
						dominant_religion ?= {
							group = religion_group:muslim
						}
						any_neighbor_location = {
							dominant_religion ?= religion:hindu
						}
					}
				}
			}
			max = 1
			save_scope_as = target_location
			dominant_religion = {
				save_scope_as = target_religion
			}
		}
		scope:target_location = {
			create_building_country_in_location = {
				name = {
					name = SIKH
				}
				save_scope_as = guru_country
				change_country_flag = SIKH
				change_country_tag = SIKH
				change_religion = religion:sikhism
				change_religion_for_ruler_and_family = { country = this religion = religion:sikhism }
				change_government_type = government_type:theocracy
				add_reform = government_reform:guru_leadership_reform
				change_heir_selection = heir_selection:guru_election
				set_liturgical_language = culture.dialect

				scope:target_location = {
					change_building_level_in_location = {
						building = building_type:gurgaddi
						value = 1
						owner = scope:guru_country
					}
				}

				create_international_organization = {
					type = international_organization_type:sikhism
					add_country_to_international_organization = scope:guru_country
					set_leader_country = scope:guru_country
				}
				if = {
					limit = {
						current_year > 1509
						current_year < 1540
					}
					create_character = {
						first_name = name_nanak
						religion = religion:sikhism
						culture = culture:punjabi
						estate = estate_type:clergy_estate
						birth_date = 1469.4.15
						birth_location = location:jaranwala #Rāi Bhoi Kī Talvaṇḍī, Nankana Sahib
						adm = { 60 90 }
						dip = { 60 90 }
						mil = { 60 90 }
						religious_figure = religious_figure:guru
						script = guru_nanak
						save_scope_as = target_character
					}
				}
				else = {
					create_character = {
						religion = religion:sikhism
						estate = estate_type:clergy_estate
						birth_location = scope:target_location
						culture = scope:target_location.dominant_culture
						min_age = 45
						adm = { 60 90 }
						dip = { 60 90 }
						mil = { 60 90 }
						religious_figure = religious_figure:guru
						script = guru_nanak
						save_scope_as = target_character
					}
				}
				set_new_ruler = scope:target_character
				scope:target_character = {
					make_saint_in_character_religion = scope:guru_country
				}
			}
		}
	}

	option = {
		name = sikhism.1.a
		show_as_tooltip = {
			religion:sikhism = { enable_religion = yes }
			scope:target_location = {
				change_building_level_in_location = {
					building = building_type:gurgaddi
					value = 1
					owner = scope:guru_country
				}
			}
		}
		custom_tooltip = {
			text = lots_of_sikhs
			lots_of_sikhs = yes
			scope:target_location = {
				every_pop = {
					limit = {
						owner = root
						OR = {
							religion = religion:hindu
							religion.group = religion_group:muslim
						}
					}
					split_pop = {
						fraction = 0.66
						religion = religion:sikhism
					}
				}
			}
		}
	}

	option = {
		name = sikhism.1.b
		show_as_tooltip = {
			religion:sikhism = { enable_religion = yes }
			scope:target_location = {
				change_building_level_in_location = {
					building = building_type:gurgaddi
					value = 1
					owner = scope:guru_country
				}
			}
		}
		change_religion = religion:sikhism
		change_religion_for_ruler_and_family = { country = ROOT religion = religion:sikhism }
		add_stability = stability_extreme_penalty
		international_organization:sikhism = {
			add_country_to_international_organization = root
		}
		custom_tooltip = {
			text = lots_of_sikhs
			lots_of_sikhs = yes
			scope:target_location = {
				every_pop = {
					limit = {
						OR = {
							religion = religion:hindu
							religion.group = religion_group:muslim
						}
					}
					split_pop = {
						fraction = 1.0
						religion = religion:sikhism
					}
				}
			}
		}
	}

	option = {
		name = sikhism.1.c
		trigger = {
			is_ai = no
		}
		show_as_tooltip = {
			religion:sikhism = { enable_religion = yes }
			scope:target_location = {
				change_building_level_in_location = {
					building = building_type:gurgaddi
					value = 1
					owner = scope:guru_country
				}
			}
		}
		change_player = scope:guru_country
		custom_tooltip = {
			text = lots_of_sikhs
			scope:target_location = {
				every_pop = {
					limit = {
						OR = {
							religion = religion:hindu
							religion.group = religion_group:muslim
						}
					}
					split_pop = {
						fraction = 1.0
						religion = religion:sikhism
					}
				}
			}
		}
	}
}

		}

			country_exists = c:SIKH
		}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		ordered_owned_location = {
			order_by = population
			max = 1
			save_scope_as = target_location
		}

				name = {
					name = SIKH
				}

					}

					}
				}
			}

				building = building_type:gurgaddi

				owner = scope:guru_country
			}
		}
	}

		show_as_tooltip = {
			scope:target_location = {
				change_building_level_in_location = {
