innovativeness_events.4 = {
	type = country_event
	title = innovativeness_events.4.title
	desc = innovativeness_events.4.desc
	image = "gfx/interface/illustrations/institutions/scientific_revolution.dds"
	trigger = {
		
		any_neighbor_country = {
			num_of_advances_researched >= root.num_of_advances_researched
		}
	}


	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				limit = {
					num_of_advances_researched >= root.num_of_advances_researched
				}
				save_scope_as = inventor_country
			}
		}
	}

	option = {
		name = innovativeness_events.4.a
		add_research_progress = research_progress_mild_bonus
		scope:inventor_country = {
			add_opinion = { target = root modifier = harbored_inventor }
			every_related_country = {
				type = alliance
				limit = {
					NOT = { this = root }
				}
				add_opinion = { target = root modifier = harbored_inventor }
			}
		}
	}
	option = {
		name = innovativeness_events.4.b
		add_prestige = prestige_mild_bonus
		scope:inventor_country = {
			add_opinion = { target = root modifier = turned_away_inventor }
		}
	}
}

innovativeness_events.16 = { # In Search of Kidney Stones
	type = country_event

	title = innovativeness_events.16.t
	desc = innovativeness_events.16.desc
	image = "gfx/interface/illustrations/institutions/scientific_revolution.dds"
	

	trigger = {
		at_war = no
		OR = {
			capital.continent = continent:europe
			capital.continent = continent:asia
			capital.sub_continent = sub_continent:north_africa
		}
		societal_value:traditionalist_vs_innovative < 0
		NOT = { has_embraced_institution = institution:scientific_revolution }
	}

	option = { # I want gold from lead!
		name = innovativeness_events.16.a
		custom_tooltip = innovativeness_events.16.event_notifier

		change_gold_effect = { scale = -5 }
		save_scope_value_as = {
			name = innovativeness_events_16_choice
			value = flag:a
		}
		trigger_event_silently = {
			id = innovativeness_events.17
			months = { 4 6 }
		}
	}

	option = { # An elixir of life!
		name = innovativeness_events.16.b
		custom_tooltip = innovativeness_events.16.event_notifier

		change_gold_effect = { scale = -4 }
		save_scope_value_as = {
			name = innovativeness_events_16_choice
			value = flag:b
		}
		trigger_event_silently = {
			id = innovativeness_events.17
			months = { 4 6 }
		}
	}

	option = { # Search high and low for the philosopher's stone!
		name = innovativeness_events.16.c
		custom_tooltip = innovativeness_events.16.event_notifier

		change_gold_effect = { scale = -24 }
		trigger_event_silently = {
			id = innovativeness_events.17
			months = { 4 6 }
		}
	}

	option = { # You will get no money from le moi good messieur!
		name = innovativeness_events.16.d
		add_prestige = prestige_mild_penalty
		change_societal_value = { type = traditionalist_vs_innovative value = societal_value_tiny_move_to_left }
	}
}

innovativeness_events.17 = { # In Search of Kidney Stones - Outcome
	type = country_event

	title = innovativeness_events.16.t
	desc = innovativeness_events.17.desc
	image = "gfx/interface/illustrations/institutions/scientific_revolution.dds"

	option = {
		name = innovativeness_events.17.a

		if = {
			limit = {
				exists = scope:innovativeness_events_16_choice 
				scope:innovativeness_events_16_choice = flag:a 
			}
			capital.market = {
				add_goods_supply = {
					goods = goods:goods_gold
					amount = market_goods_mild_gain
				}
				add_goods_supply = {
					goods = goods:silver
					amount = market_goods_mild_gain
				}
				add_goods_supply = {
					goods = goods:iron
					amount = market_goods_severe_gain
				}
				add_goods_supply = {
					goods = goods:lead
					amount = market_goods_severe_gain
				}
			}
		}
		else_if = {
			limit = {
				exists = scope:innovativeness_events_16_choice 
				scope:innovativeness_events_16_choice = flag:b 
			}
			capital.market = {
				add_goods_supply = {
					goods = goods:medicaments
					amount = market_goods_extreme_gain
				}
				add_goods_supply = {
					goods = goods:gems
					amount = market_goods_severe_gain
				}
			}
			add_research_progress = research_progress_severe_bonus
		}
		else = {
			add_research_progress = research_progress_extreme_bonus
			capital.market = {
				add_goods_supply = {
					goods = goods:saltpeter
					amount = market_goods_extreme_gain
				}
				add_goods_supply = {
					goods = goods:goods_gold
					amount = market_goods_mild_gain
				}
			}
		}
	}
}
