namespace = colonial_policies

colonial_policies.10 = {
	type = country_event
	title = colonial_policies.10.title
	desc = colonial_policies.10.desc
	trigger = {
		has_policy = settled_colonies
		NOT = {	has_country_modifier = colonial_enthusiasm	}
		any_colonial_charter = {
			count > 0
		}
	}
	illustration_tags = {
        10 = happy
        10 = exterior
    }
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
	}
	option = {
		name = colonial_policies.10.a
		add_country_modifier = { modifier = colonial_enthusiasm years = 10 mode = add_and_extend }  
	}
}

colonial_policies.11 = {
	hide_portraits = yes
	type = country_event
	title = colonial_policies.11.title
	desc = colonial_policies.11.desc
	trigger = {
		at_war = yes
		any_subject = {
			is_subject_type = colonial_nation
		}
		
	}
	
	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		random_subject = {
			limit = {
				is_subject_type = colonial_nation
			}
			save_scope_as = colony
		}
	}
	

	option = {
		name = colonial_policies.11.a
		
		add_migration = {
			owner = this
			from = capital.province_definition
			to = scope:colony.capital.province_definition
			type = pop_type:peasants
			amount = 0.2
			months = 240
		}	
		
	}
	option = {
		name = colonial_policies.11.b
		add_war_exhaustion = war_exhaustion_weak_penalty
	}
}
