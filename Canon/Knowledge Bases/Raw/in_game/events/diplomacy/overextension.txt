namespace = overextension
overextension.1 = {
	type = country_event
	title = overextension.1.title
	desc = overextension.1.desc
	trigger = {
		at_war = yes
		has_countries_with_coalition_grade_antagonism = yes
		monthly_manpower > 0
	}
	illustration_tags = {
        10 = regular
        10 = exterior
    }
	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
	}
	option = {
		name = overextension.1.a
		add_war_exhaustion = war_exhaustion_weak_penalty
		add_manpower = { value = root.monthly_manpower multiply = -0.50 }
	}
}

overextension.6 = {
	type = country_event
	title = overextension.6.title
	desc = overextension.6.desc
	trigger = {
		has_countries_with_coalition_grade_antagonism = yes

		any_market_with_merchants = {
			NOT = {
				has_merchant_power = {
					country = root
					key = merchant_recalled
				}
			}
		}
	}
	illustration_tags = {
        10 = angry
        10 = interior
    }
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }
	}
	option = {
		name = overextension.6.a
		random_market_with_merchants = {
			limit = {
				NOT = {
					has_merchant_power = {
						country = root
						key = merchant_recalled
					}
				}
			}
			add_merchant_power = {
				country = root
				key = merchant_recalled
				months = 120
				power = merchant_power_weak_penalty
			}
		}
	}
}

overextension.10 = {
	hide_portraits = yes
	type = country_event
	title = overextension.10.title
	desc = overextension.10.desc
	trigger = {
		has_countries_with_coalition_grade_antagonism = yes
		any_cabinet_character = {
			count > 0
		}
	}
	
	illustration_tags = {
        10 = angry
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		random_cabinet_character = {
			save_scope_as = minister
		}
	}

	option = {
		name = overextension.10.a
		remove_from_cabinet = scope:minister
		scope:minister = { add_character_modifier = { modifier = refuses_to_serve_in_cabinet years = -1 mode = add_and_extend }	}
	}
}
