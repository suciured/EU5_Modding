russian_principality_events.2 = {
	type = country_event
	title = russian_principality_events.2.title
	desc = russian_principality_events.2.desc
	trigger = {
		is_russian_principality = yes
		num_of_non_rural < num_locations
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:peasants_estate background = estate_type:peasants_estate }
		hidden_effect = {
			random_owned_rural_location = {
				save_scope_as = agricultural_intensification_province
			}
		}
	}

	option = {
		name = russian_principality_events.2.a
		scope:agricultural_intensification_province = {
			change_development = development_weak_bonus
		}
	}
}

russian_principality_events.4 = {
	type = country_event
	title = russian_principality_events.4.title
	desc = russian_principality_events.4.desc
	trigger = {
		is_russian_principality = yes
		NOT = { is_member_of_international_organization = international_organization:tatar_yoke }
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		random_owned_location = {
			save_scope_as = target_location
		}
	}

	#land is true wealth
	option = {
		name = russian_principality_events.4.a
		scope:target_location = {
			change_prosperity = prosperity_severe_bonus
		}
	}

	#we need gold for the treasury
	option = {
		name = russian_principality_events.4.b
		change_gold_effect = { scale = 2 }
	}

	#use this as leverage
	option = {
		name = russian_principality_events.4.c
		trigger = {
			has_countries_with_timed_antagonism = yes
		}
		drop_antagonism_bomb = {
			target = scope:target_location
			modifier = antagonism_bought_relief
		}
	}
}

russian_principality_events.9 = {
	type = country_event
	title = russian_principality_events.9.title
	desc = russian_principality_events.9.desc
	trigger = {
		is_russian_principality = yes
		NOT = { is_member_of_international_organization = international_organization:tatar_yoke }
		"religion_percentage_in_country(religion:sunni)" > 0
		religion = religion:orthodox
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
		ordered_province = {
			order_by = "religion_percentage(religion:sunni)"
			max = 1
			save_scope_as = tatar_province
		}
	}

	option = {
		name = russian_principality_events.9.a
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_penalty }
		
		scope:tatar_province = {
			every_location_in_province = {
				
				every_pop = {
					limit = {
						religion = religion:sunni
					}
					split_pop = {
						fraction = 0.10
						religion = religion:orthodox
					}
				}		
			}
		}
	}
	option = {
		name = russian_principality_events.9.b
		add_religious_influence = religious_influence_mild_bonus
		add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
	}
}
