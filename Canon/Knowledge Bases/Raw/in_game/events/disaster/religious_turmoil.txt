religious_turmoil.3 = {
	type = country_event
	category = disaster_event
	title = religious_turmoil.3.title
	desc = religious_turmoil.3.desc
	trigger = {
		societal_value:spiritualist_vs_humanist < -10
	}
	image = "gfx/interface/illustrations/disaster/religious_turmoil.dds"

	option = {
		name = religious_turmoil.3.a

		custom_tooltip = religious_turmoil.3.a.tt
		hidden_effect = {
			ordered_owned_location = {
				order_by = population
				limit = {
					local_religious_unity < 1.0
				}
				max = {
					value = 1
					add = {
						value = root.num_provinces
						multiply = 0.2
					}
				}
				check_range_bounds = no
				every_pop = {
					limit = {
						owner = root
						religion != root.religion
					}
					add_pop_satisfaction = pop_satisfaction_mild_penalty
				}
			}
		}
	}
	option = {
		name = religious_turmoil.3.b
		change_gold_effect = { scale = -6 }
	}
}

religious_turmoil.5 = {
	type = country_event
	category = disaster_event
	title = religious_turmoil.5.title
	desc = religious_turmoil.5.desc
	trigger = {
		societal_value:spiritualist_vs_humanist > 10
	}
	image = "gfx/interface/illustrations/disaster/religious_turmoil.dds"
	immediate = {
		if = {
			limit = {
				religion = religion:catholic
			}
			religion:lutheran = {
				save_scope_as = heretic_religion
			}
		}

		if = {
			limit = {
				religion != religion:catholic
			}
			religion:catholic = {
				save_scope_as = heretic_religion
			}
		}
	}

	option = {
		name = religious_turmoil.5.a
		add_stability = stability_mild_penalty
	}
	option = {
		name = religious_turmoil.5.b

		ordered_owned_location = {
			order_by = population
			max = {
				value = 1
				add = {
					value = root.num_provinces
					multiply = 0.2
				}
			}
			check_range_bounds = no
			every_pop = {
				limit = {
					owner = root
					religion != root.religion
				}
				add_pop_satisfaction = pop_satisfaction_mild_penalty
			}
		}
	}
}
