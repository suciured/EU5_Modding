theocratic_flavor.1 = {
	type = country_event
	title = theocratic_flavor.1.title
	desc = theocratic_flavor.1.desc
	trigger = {
		government_type = government_type:theocracy
		NOT = {	has_country_modifier = its_not_gold	}
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
	}
	
	option = {
		name = theocratic_flavor.1.a
		add_gold = 80.00
	}
	option = {
		name = theocratic_flavor.1.b
		add_country_modifier = { modifier = its_not_gold years = 10 mode = add_and_extend }  
	}
}

theocratic_flavor.9 = {
	type = country_event
	title = theocratic_flavor.9.title
	desc = theocratic_flavor.9.desc
	trigger = {
		government_type = government_type:theocracy

		any_owned_location = {
			NOT = {
				dominant_religion = root.religion
			}
		}
	}

	illustration_tags = {
        10 = happy
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
	
		ordered_owned_location = {
			order_by = population
			limit = {
				NOT = {
					dominant_religion = root.religion
				}
			}
			max = {
				value = 1
				add = {
					value = root.num_provinces
					multiply = 0.2
				}
			}
			check_range_bounds = no

			add_to_list = theocratic_flavor_9_list
			every_pop = {
				limit = {}
				split_pop = {
					fraction = 0.10
					religion = root.religion
				}
			}
		}
	}
	
	option = {
		name = theocratic_flavor.9.a
		if = {
			limit = {
				religion = religion:catholic
			}
			add_religious_influence = religious_influence_mild_bonus
		}
		
		random_in_list = {
			list = theocratic_flavor_9_list

			ordered_pop = {
				limit = { religion != root.religion }
				order_by = pop_size
				max = 4
				check_range_bounds = no
				split_pop = {
					fraction = { 0.08 0.12 }
					religion = root.religion
				}
			}
		}
	
	}
}
