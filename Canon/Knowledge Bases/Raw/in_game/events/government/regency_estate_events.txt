namespace = regency_estate_events
regency_estate_events.1 = {
	type = country_event
	title = regency_estate_events.1.title
	desc = regency_estate_events.1.desc
	
	
	trigger = {
		has_regent = yes
		regency_type = regency_type:nobles_estate_regency
		"estate_power(estate_type:burghers_estate)" > "estate_power(estate_type:nobles_estate)"
	}

	immediate = {
		regent = { save_scope_as = target_regent }
	}

	option = {
		name = regency_estate_events.1.a
		reset_regency = yes
	}
	option = {
		name = regency_estate_events.1.b
		add_estate_satisfaction = { type = estate_type:burghers_estate 	value = estate_satisfaction_extreme_penalty }
		add_estate_satisfaction = { type = estate_type:nobles_estate 	value = estate_satisfaction_mild_bonus }
	}
}

regency_estate_events.9 = {
	type = country_event
	title = regency_estate_events.9.title
	desc = regency_estate_events.9.desc
	trigger = {
		has_heir = yes
		heir = {
			owner = ROOT
			age_in_years > 6 
			age_in_years < 16
		}
	
		has_regent = yes

		regent = {
			total_abilities > 200
			location = ROOT.heir.location
		}

		OR = {
			regency_type = regency_type:nobles_estate_regency
			regency_type = regency_type:burghers_estate_regency
			regency_type = regency_type:clergy_estate_regency
			regency_type = regency_type:peasants_estate_regency
		}
	}

	immediate = {
		regent = { save_scope_as = target_regent }
		heir = { save_scope_as = target_heir }
	}

	option = {
		name = regency_estate_events.9.a
		scope:target_heir = {
			random_list = {
				30 = {
					add_adm = 10
				}
				30 = {
					add_dip = 10
				}
				30 = {
					add_mil = 10
				}
				10 = {
					add_adm = 5
					add_dip = 5
					add_mil = 5
				}
			}
		}
	}
}
