namespace = sailor_event
sailor_event.1 = {
	type = country_event
	title = sailor_event.1.title
	desc = sailor_event.1.desc
	trigger = {
		num_of_ports >= 1
		navy_tradition >= 10.00
		NOT = {
			sailors_percentage >= 0.90
		}
		navy_size > 0
		monthly_sailors > 0.1
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
	}
	
	option = {
		name = sailor_event.1.a
		add_sailors = { value = root.monthly_sailors multiply = 1.50 }
	}
}

sailor_event.5 = {
	type = country_event
	title = sailor_event.5.title
	desc = sailor_event.5.desc
	trigger = {
		num_of_non_rural > 5
		num_of_ports > 0
		num_of_ports < 3
		navy_size > 0

		any_owned_non_rural_location = {
			is_port = no
			development > 5
		}
		any_owned_location = {
			is_port = yes
			development < 90
		}
	}

	illustration_tags = {
        10 = regular
        10 = exterior
    }

	immediate = {
		event_illustration_poptype_effect = { foreground = pop_type:soldiers background = pop_type:soldiers }
	}
	
	option = {
		name = sailor_event.5.a
		random_owned_location = {
			limit = {
				is_port = yes
				development < 90
			}
			change_development  = development_mild_bonus
		}
		random_owned_non_rural_location = {
			limit = {
				is_port = no
				development > 5
			}
			change_development  = development_weak_penalty
		}
		change_gold_effect = { scale = -2.40 }
	}
	option = {
		name = sailor_event.5.b
		add_prestige = prestige_weak_bonus
	}
}
