# CULTURE
# root is the culture
# scope:target is the country to filter

culture_present_in_country = {
	scope = culture
	tag = culture
	group = 1
    exclusive_group = yes
	enabled_at_start = yes
	trigger = {
		scope:target = {
			culture_percentage_in_country:root > 0
		}
	}
}

culture_not_present_in_country = {
	scope = culture
	tag = culture
	group = 1
    exclusive_group = yes
	invert = yes
	trigger = {
		scope:target = {
			culture_percentage_in_country:root > 0
		}
	}
}
